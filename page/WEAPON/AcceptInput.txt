<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>A workaround for weapons created by maps not taking into account spawnflags.
</p>
<pre><code data-language="lua">function SWEP:AcceptInput( name, activator, caller, data )

	-- Check for input and spawnflag
	if ( name == &quot;ConstraintBroken&quot; &amp;&amp; self:HasSpawnFlags( 1 ) ) then

		-- Freeze the weapon
		local phys = self:GetPhysicsObject()
		if ( IsValid( phys ) ) then phys:EnableMotion( false ) end
	
		-- Remove the spawnflag so it doesn't freeze the weapon when it is dropped
		local newflags = bit.band( self:GetSpawnFlags(), bit.bnot( 1 ) )
		self:SetKeyValue( &quot;spawnflags&quot;, newflags )
	end
end</code></pre>
