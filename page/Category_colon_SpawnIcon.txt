<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a DFrame with a SpawnIcon inside.
</p>
<pre><code data-language="lua">local Panel = vgui.Create( &quot;DFrame&quot; ) -- Main Frame
Panel:SetPos( 200, 200 )
Panel:SetSize( 200, 200 )
Panel:SetTitle( &quot;Spawn Icon Test&quot; )
Panel:SetVisible( true )
Panel:SetDraggable( false )
Panel:ShowCloseButton( true )
Panel:MakePopup()

local SpawnI = vgui.Create( &quot;SpawnIcon&quot; , Panel ) -- SpawnIcon
SpawnI:SetPos( 75, 75 )
SpawnI:SetModel( &quot;models/props_borealis/bluebarrel001.mdl&quot; ) -- Model we want for this spawn icon</code></pre>
