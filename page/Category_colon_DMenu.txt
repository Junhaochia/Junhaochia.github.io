<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a DMenu with a submenu.
</p>
<pre><code data-language="lua">local Menu = DermaMenu()

-- Add a simple option.
Menu:AddOption( &quot;Simple option&quot; )

-- Simple option, but we're going to add an icon
local btnWithIcon = Menu:AddOption( &quot;Option with icon&quot; )
btnWithIcon:SetIcon( &quot;icon16/bug.png&quot; )	-- Icons are in materials/icon16 folder

-- Adds a simple line spacer
Menu:AddSpacer()

-- Add a submenu
local SubMenu = Menu:AddSubMenu( &quot;A Sub Menu&quot; )
SubMenu:AddOption( &quot;Sub Option #1&quot; ):SetIcon( &quot;icon16/group.png&quot; )

-- Add a submenu with icon
local Child, Parent = Menu:AddSubMenu( &quot;A Sub Menu with Icon&quot; )
Parent:SetIcon( &quot;icon16/arrow_refresh.png&quot; )
Child:AddOption( &quot;Sub Option #2&quot; ):SetIcon( &quot;icon16/group.png&quot; )

-- Open the menu
Menu:Open()</code></pre>
