<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>How to create a basic IVideoWriter, and how to use it
</p>
<pre><code data-language="lua">local config = {
	container = &quot;webm&quot;,
	video = &quot;vp8&quot;,
	audio = &quot;vorbis&quot;,
	quality = 50,
	bitrate = 200,
	fps = 30,
	lockfps = 30,
	name = &quot;Test&quot;,
	width = 1280,
	height = 720
}
local iVideoWriter = video.Record( config )
iVideoWriter:SetRecordSound( true )

local function Record()
	iVideoWriter:AddFrame( FrameTime(), true )
end

function StartRecording()
	hook.Add( &quot;DrawOverlay&quot;, &quot;Record&quot;, Record )
end

function StopRecording()
	hook.Remove( &quot;DrawOverlay&quot;, &quot;Record&quot; )
	iVideoWriter:Finish()
end</code></pre>
