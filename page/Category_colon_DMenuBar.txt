<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a DMenuBar in a DFrame.
</p>
<pre><code data-language="lua">local Frame = vgui.Create( &quot;DFrame&quot; )
Frame:SetSize( 300, 200 )
Frame:Center()
Frame:MakePopup()

local MenuBar = vgui.Create( &quot;DMenuBar&quot;, Frame )
MenuBar:DockMargin( -3, -6, -3, 0 ) --corrects MenuBar pos

local M1 = MenuBar:AddMenu( &quot;File&quot; )
M1:AddOption(&quot;New&quot;, function() Msg(&quot;Chose File:New\n&quot;) end):SetIcon(&quot;icon16/page_white_go.png&quot;)
M1:AddOption(&quot;Open&quot;, function() Msg(&quot;Chose File:Open\n&quot;) end):SetIcon(&quot;icon16/folder_go.png&quot;)

local M2 = MenuBar:AddMenu(&quot;Edit&quot;)
M2:AddOption(&quot;Copy&quot;, function() Msg(&quot;Chose Edit:Copy\n&quot;) end)

local M3 = MenuBar:AddMenu(&quot;Help&quot;)
M3:AddOption(&quot;About&quot;, function() Msg(&quot;Chose Help:About\n&quot;) end)</code></pre>

<h1 class="first_example"> <span class="mw-headline" id="Examples_2">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example_2">Example</span></h2>
<p>A sub menu example
</p>
<pre><code data-language="lua">local frame = vgui.Create( &quot;DFrame&quot; )
frame:SetSize( 300, 200 )
frame:Center()
frame:MakePopup()

local menuBar = vgui.Create( &quot;DMenuBar&quot;, frame )
menuBar:DockMargin( -3, -6, -3, 0 )

local option1 = menuBar:AddMenu( &quot;Option 1&quot; )
option1:AddOption( &quot;Dummy option&quot;, function() MsgN( &quot;Chose Dummy option&quot; ) end )

local sub = option1:AddSubMenu( &quot;Sub Menu&quot; )
sub:SetDeleteSelf( false ) -- Necessary so the sub menu is not automatically removed on close
for i = 0, 5 do
	sub:AddOption( &quot;Option &quot; .. i, function() MsgN( &quot;Chose sub menu option &quot; .. i ) end )
end</code></pre>
