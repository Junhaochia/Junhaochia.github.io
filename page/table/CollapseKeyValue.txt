<h1> <span class="mw-headline" id="Description">Description</span></h1>

<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Example usage
</p>
<pre><code data-language="lua">local output = table.CollapseKeyValue( {
	{ Key = &quot;mykey1&quot;, Value = &quot;myvalue1&quot; },
	{ Key = 123, Value = 1345 },
	{ Key = 1345, Value = &quot;myvalue1&quot; },
} )</code></pre>
<p><b>Output:</b>
</p>
<div style="margin:2em;"><pre><code data-language="lua">local output = {
	[ &quot;mykey1&quot; ] = &quot;myvalue1&quot;,
	[ 123 ] = 1345,
	[ 1345 ] = &quot;myvalue1&quot;,
}</code></pre></div>
