<h1> <span class="mw-headline" id="Description">Description</span></h1>
<p>Sets lighting mode when rendering something.
</p>
<table cellspacing="0" class="gmodwiki_note">
<tr>
<td>NOTE
</td>
<td>
<p><b>Do not forget to restore the default value</b> to avoid unexpected behavior, like the world and the HUD/UI being affected
</p>
</td></tr></table>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_number" title="Category:number">number</a> Mode</span>
</p>
<div style="margin-left: 32px;">
<p>Lighting render mode
</p><p>Possible values are:
</p>
<ul><li> 0 - Default
</li><li> 1 - Total fullbright, similar to <i>mat_fullbright 1</i> but excluding some weapon view models
</li><li> 2 - Increased brightness(?), models look fullbright</div>
</li></ul>
</div>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Draws a fullbright quad on 2D skybox
</p>
<pre><code data-language="lua">local MATERIAL = Material(&quot;skybox/trainup&quot;)

hook.Add(&quot;PostDraw2DSkyBox&quot;, &quot;Quaddrawer&quot;, function()
	render.OverrideDepthEnable( true, false )
	render.SetLightingMode(2)

	cam.Start3D(Vector(0, 0, 0), EyeAngles())
		render.SetMaterial(MATERIAL)
		render.DrawQuadEasy(Vector(200,0,0), Vector(-1,0,0), 64, 64, Color(255,255,255), 180)
	cam.End3D()

	render.OverrideDepthEnable( false, false )
	render.SetLightingMode(0)
end)</code></pre>
<p><br />
</p>
<h1 class="first_example"> <span class="mw-headline" id="Examples_2">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example_2">Example</span></h2>
<p>Display everything the same way as when you set <i>mat_fullbright</i> to 1.
</p>
<pre><code data-language="lua">local LightingModeChanged = false
hook.Add( &quot;PreRender&quot;, &quot;fullbright&quot;, function()
	render.SetLightingMode( 1 )
	LightingModeChanged = true
end )

local function EndOfLightingMod()
	if LightingModeChanged then
		render.SetLightingMode( 0 )
		LightingModeChanged = false
	end
end
hook.Add( &quot;PostRender&quot;, &quot;fullbright&quot;, EndOfLightingMod )
hook.Add( &quot;PreDrawHUD&quot;, &quot;fullbright&quot;, EndOfLightingMod )</code></pre>
