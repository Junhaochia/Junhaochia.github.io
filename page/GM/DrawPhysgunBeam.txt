<h1> <span class="mw-headline" id="Description">Description</span></h1>
<p>Allows you to override physgun beam drawing.
</p>
<table class="gmodwiki_note bug_note">
<tr>
<td>BUG
</td>
<td>
<p>This is still called when physgun_drawbeams is disabled.

</p>

<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Example code that will draw a direct line from the physgun to the target.
</p>
<pre><code data-language="lua">hook.Add( &quot;DrawPhysgunBeam&quot;, &quot;test&quot;, function( ply, wep, enabled, target, bone, deltaPos )

	-- Draw any physgun effects here that are not the beam.

	-- Not &quot;firing&quot; the physgun? Don't draw anything.
	if (&#160;!enabled ) then return false end

	local clr = Color( 255, 0, 0 )

	-- White when not &quot;firing&quot; physgun, this will not work with the &quot;if&quot; above
	if (&#160;!enabled ) then clr = Color( 255, 255, 255, 255 ) end

	local hitpos = ply:GetEyeTrace().HitPos
	if ( IsValid( target ) ) then
		local mt = target:GetBoneMatrix( bone )
		if ( target:TranslatePhysBoneToBone( bone ) &gt;= 0 ) then
			mt = target:GetBoneMatrix( target:TranslatePhysBoneToBone( bone ) )
		end

		hitpos = LocalToWorld( deltaPos, Angle( 0, 0, 0 ), mt:GetTranslation(), mt:GetAngles() )
	end

	local srcPos = wep:GetAttachment( 1 ).Pos
	if (&#160;!ply:ShouldDrawLocalPlayer() ) then
		srcPos = ply:GetViewModel():GetAttachment( 1 ).Pos
	end

	render.DrawLine( srcPos, hitpos, clr )

	return false -- Hide original physics gun beam

end )</code></pre>
