<h1> <span class="mw-headline" id="Description">Description</span></h1>
<p>Called when it's time to populate the context menu menu bar at the top.
</p>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_Panel" title="Category:Panel">Panel</a> menubar</span>
</p>
<div style="margin-left: 32px;">
The <a href="/page/Category_colon_DMenuBar" title="Category:DMenuBar">DMenuBar</a> itself.</div>
</div>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Example usage of the hook
</p>
<pre><code data-language="lua">hook.Add( &quot;PopulateMenuBar&quot;, &quot;My_MenuBar&quot;, function( menubar )

	local m = menubar:AddOrGetMenu( &quot;Test&quot; )

	m:AddCVar( &quot;Item 1&quot;, &quot;console_var1&quot;, &quot;1&quot;, &quot;0&quot; )

	m:AddSpacer()

	m:AddCVar( &quot;Item 2&quot;, &quot;console_var2&quot;, &quot;0&quot;, &quot;100&quot; )

	m:AddCVar( &quot;Check console&quot;, &quot;console_var3&quot;, &quot;1&quot;, &quot;0&quot;, function() print(&quot;I was clicked!&quot;) end )


	local submenu = m:AddSubMenu( &quot;Submenu&quot; )

	submenu:SetDeleteSelf( false )
	submenu:AddCVar( &quot;No password&quot;, &quot;password&quot;, &quot;&quot; )
	submenu:AddSpacer()

	submenu:AddCVar( &quot;Password: test1&quot;, &quot;password&quot;, &quot;test1&quot; )
	submenu:AddCVar( &quot;Password: lolno&quot;, &quot;password&quot;, &quot;lolno&quot; )

end )</code></pre>
