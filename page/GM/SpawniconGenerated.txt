<h1> <span class="mw-headline" id="Description">Description</span></h1>
<p>Called when spawn icon is generated.
</p>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_string" title="Category:string">string</a> lastmodel</span>
</p>
<div style="margin-left: 32px;">
File path of previously generated model.</div>
</div>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments_2">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_string" title="Category:string">string</a> imagename</span>
</p>
<div style="margin-left: 32px;">
File path of the generated icon.</div>
</div>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments_3">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_number" title="Category:number">number</a> modelsleft</span>
</p>
<div style="margin-left: 32px;">
Amount of models left to generate.</div>
</div>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>That's how it is used in <b>garrysmod/lua/includes/gui/icon_progress.lua</b> for show progress of generating icons
</p>
<pre><code data-language="lua">local g_Progress = nil

hook.Add( &quot;SpawniconGenerated&quot;, &quot;SpawniconGenerated&quot;, function( lastmodel, imagename, modelsleft )

	if (&#160;!IsValid( g_Progress ) ) then
	
		g_Progress = vgui.Create( &quot;DPanel&quot; )
		g_Progress:SetSize( 64+10, 64+10+20 )
		g_Progress:SetBackgroundColor( Color( 0, 0, 0, 100 ) )
		g_Progress:SetDrawOnTop( true )
		g_Progress:DockPadding( 5, 0, 5, 5 )
		g_Progress.Think = function()
		
			if ( SysTime() - g_Progress.LastTouch &lt; 3 ) then return end
			
			g_Progress:Remove()
			g_Progress.LastTouch = SysTime()
		
		end
		

		
		local label = g_Progress:Add( &quot;DLabel&quot; )
		label:Dock( BOTTOM )
		label:SetText( &quot;remaining&quot; )
		label:SetTextColor( Color( 255, 255, 255, 255 ) )
		label:SetExpensiveShadow( 1, Color( 0, 0, 0, 200 ) )
		label:SetContentAlignment( 5 )
		label:SetHeight( 14 )
		label:SetFont( &quot;DefaultSmall&quot; )
		
		g_Progress.Label = g_Progress:Add( &quot;DLabel&quot; )
		g_Progress.Label:Dock( BOTTOM )
		g_Progress.Label:SetTextColor( Color( 255, 255, 255, 255 ) )
		g_Progress.Label:SetExpensiveShadow( 1, Color( 0, 0, 0, 200 ) )
		g_Progress.Label:SetContentAlignment( 5 )
		g_Progress.Label:SetFont( &quot;DermaDefaultBold&quot; )
		g_Progress.Label:SetHeight( 14 )
		
		g_Progress.icon = vgui.Create( &quot;DImage&quot;, g_Progress )
		g_Progress.icon:SetSize( 64, 64 )
		g_Progress.icon:Dock( TOP )
	
	end
	
	g_Progress.LastTouch = SysTime()
	
	imagename = imagename:Replace( &quot;materials\\&quot;, &quot;&quot; )
	imagename = imagename:Replace( &quot;materials/&quot;, &quot;&quot; )
		
	g_Progress.icon:SetImage( imagename )
		
	g_Progress:AlignRight( 10 )
	g_Progress:AlignBottom( 10 )
	
	g_Progress.Label:SetText( modelsleft )

end )</code></pre>
