<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>From base/gamemode/variable_edit.lua
</p>
<pre><code data-language="lua">function GM:VariableEdited( ent, ply, key, val, editor )
    if (&#160;!IsValid( ent ) ) then return end
    if (&#160;!IsValid( ply ) ) then return end
    local CanEdit = hook.Run( &quot;CanEditVariable&quot;, ent, ply, key, val, editor )
    if (&#160;!CanEdit ) then return end
    ent:EditValue( key, val )
end</code></pre>
