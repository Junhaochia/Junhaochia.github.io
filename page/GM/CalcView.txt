<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Draws the LocalPlayer and sets the view behind.
</p>
<pre><code data-language="lua">local function MyCalcView(ply, pos, angles, fov)
    local view = {}
    view.origin = pos-(angles:Forward()*100)
    view.angles = angles
    view.fov = fov
    view.drawviewer = true
 
    return view
end
 
hook.Add( &quot;CalcView&quot;, &quot;MyCalcView&quot;, MyCalcView )</code></pre>
