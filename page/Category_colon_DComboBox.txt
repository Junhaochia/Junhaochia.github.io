<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a Combo Box.
</p>
<pre><code data-language="lua">local frame = vgui.Create( &quot;DFrame&quot; )
frame:SetSize( 300, 250 )
frame:Center()
frame:MakePopup()

local DComboBox = vgui.Create( &quot;DComboBox&quot;, frame )
DComboBox:SetPos( 5, 30 )
DComboBox:SetSize( 100, 20 )
DComboBox:SetValue( &quot;options&quot; )
DComboBox:AddChoice( &quot;option A&quot; )
DComboBox:AddChoice( &quot;option B&quot; )
DComboBox:AddChoice( &quot;option C&quot; )
DComboBox.OnSelect = function( self, index, value )
	print( value ..&quot; was selected at index &quot; .. index )
end</code></pre>
<p><b>Output:</b>
</p>
<div style="margin:2em;"><br />
<p><a href="/page/File:PictureA.png" class="image"><img alt="PictureA.png" src="/images/1/17/PictureA.png" width="104" height="24" /></a>
<a href="/page/File:DComboBox2.png" class="image"><img alt="DComboBox2.png" src="/images/5/53/DComboBox2.png" width="104" height="90" /></a>
<a href="/page/File:DComboBox3.png" class="image"><img alt="DComboBox3.png" src="/images/7/73/DComboBox3.png" width="104" height="24" /></a>
</p>
option B was selected!</div>
<h1 class="first_example"> <span class="mw-headline" id="Examples_2">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example_2">Example</span></h2>
<p>Create a Combo Box with all players
</p>
<pre><code data-language="lua">local comboBox = vgui.Create(&quot;DComboBox&quot;)
comboBox:SetPos(5, 5)
comboBox:SetSize(100, 20)
comboBox:SetValue(&quot;All Players&quot;)

comboBox.OnSelect = function( _, _, value)
	print(value..&quot; was selected!&quot;)
end

for k,v in pairs(player.GetAll()) do
	comboBox:AddChoice(v:Name())
end</code></pre>
<p><b>Output:</b>
</p>
<div style="margin:2em;"><br />
<p><a href="/page/File:DComboBoxAllPlayers.png" class="image"><img alt="DComboBoxAllPlayers.png" src="/images/1/1e/DComboBoxAllPlayers.png" width="111" height="81" /></a>
</p>
Bot01 was selected!</div>
