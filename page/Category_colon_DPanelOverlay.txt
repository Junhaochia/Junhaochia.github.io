<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a <a href="/page/Category_colon_DPanel" title="Category:DPanel">DPanel</a> with a DPanelOverlay and adds a <a href="/page/Category_colon_DLabel" title="Category:DLabel">DLabel</a> to it
</p>
<pre><code data-language="lua">local DFrame = vgui.Create( &quot;DFrame&quot; )
DFrame:SetSize( 350, 200 ) -- Set the size of the panel
DFrame:Center()
DFrame:MakePopup()

local DPanel = vgui.Create( &quot;Panel&quot;, DFrame )
DPanel:Dock( FILL )

local DLabel = vgui.Create( &quot;DLabel&quot;, DPanel )
DLabel:SetPos( 10, 10 ) -- Set the position of the label
DLabel:SetText( &quot;I'm a DLabel inside a DPanel that has a DPanelOverlay border!&quot; ) -- Set the text of the label
DLabel:SizeToContents() -- Size the label to fit the text in it

local DPanelOverlay = vgui.Create( &quot;DPanelOverlay&quot;, DPanel )
DPanelOverlay:SetType( 1 ) -- Sets the type of overlay to add to the DPanel
DPanelOverlay:SetColor( Color( 255, 0, 0 ) ) -- Sets the colour of the borders</code></pre>
