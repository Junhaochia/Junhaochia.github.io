<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Drawing a Tool Tip manually, then painting over it.
</p>
<pre><code data-language="lua">local DFrame = vgui.Create( &quot;DFrame&quot; )
DFrame:SetPos( 100, 100 )
DFrame:SetSize( 100, 100 )

local DTooltip = vgui.Create( &quot;DTooltip&quot; )
DTooltip:SetPos( 0, 0 )
DTooltip:SetSize( 250, 50 )
DTooltip:SetText( &quot;Manual Tooltip&quot; )
DTooltip:OpenForPanel( DFrame )
DTooltip:PositionTooltip()
DTooltip.Paint = function()
	draw.RoundedBox( 5, 0, 0, 250, 50, Color( 255, 255, 255, 255 ) )
end</code></pre>
