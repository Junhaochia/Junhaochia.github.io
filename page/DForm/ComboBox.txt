<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<pre><code data-language="lua">local frame = vgui.Create( &quot;DFrame&quot; )
frame:SetSize( ScrW() / 2, ScrH() / 2 )
frame:MakePopup()

local form = frame:Add(&quot;DForm&quot;)
form:Dock( FILL )
form:DockMargin( 5, 5, 5, 5 )

local combobox, label = form:ComboBox( &quot;test&quot;, &quot;sv_accelerate&quot; )
combobox:AddChoice( &quot;10&quot; ) -- 10 will be used as convar value
combobox:AddChoice( &quot;1000&quot;, 100 ) -- 100 will be used as convar value</code></pre>
