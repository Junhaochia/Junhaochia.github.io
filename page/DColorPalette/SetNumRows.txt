<h1> <span class="mw-headline" id="DescriptionPanelFunc">Description<span style="display: none;"><a href="/page/Form:PanelFunc" title="Form:PanelFunc">PanelFunc</a></span></span></h1>
<p>Roughly sets the number of colors that can be picked by the user. If the DColorPalette is exactly 6 rows tall, this function will set the number of colors shown per row in the palette.
</p>
<table cellspacing="0" class="gmodwiki_note">
<tr>
<td>NOTE
</td>
<td>
<p><a href="/page/Category_colon_DColorPalette" title="Category:DColorPalette">DColorPalette</a>:<a href="/page/DColorPalette/Reset" title="DColorPalette/Reset">Reset</a> or <a href="/page/Category_colon_DColorPalette" title="Category:DColorPalette">DColorPalette</a>:<a href="/page/DColorPalette/ResetSavedColors" title="DColorPalette/ResetSavedColors">ResetSavedColors</a> must be called after this function to apply changes.
</p>
</td></tr></table>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_number" title="Category:number">number</a> rows</span>
</p>
<div style="margin-left: 32px;">
Scale for the range of colors that the user can pick. Default is 8.</div>
</div>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates four DColorPalettes, each with a varying number of colors per row.
</p>
<pre><code data-language="lua">local frame = vgui.Create( &quot;DFrame&quot; ) -- Create the window
frame:SetSize( 350, 350 )
frame:SetTitle( &quot;SetNumRows Demonstration&quot; )
frame:Center()
frame:MakePopup()

local label = vgui.Create( &quot;DLabel&quot;, frame ) -- Create the help text
label:Dock( TOP )
label:SetText( &quot;Difference between SetNumRows( 4 ), SetNumRows( 8 ), SetNumRows( 16 ) and SetNumRows( 24 )&quot; )
label:SetWrap( true ) -- Enables text wrapping for lower resolutions
label:SetAutoStretchVertical( true ) -- Needed for the text to show properly

local pallette1 = vgui.Create( &quot;DColorPalette&quot;, frame ) -- First DColorPalette with 4 colors per row
pallette1:SetPos( 4, 60 )
pallette1:SetNumRows( 4 )
pallette1:SetSize( 40, 60 )
pallette1:Reset()

local pallette2 = vgui.Create( &quot;DColorPalette&quot;, frame ) -- Second DColorPalette with 8 colors per row
pallette2:SetPos( 4, 130 )
pallette2:SetNumRows( 8 )
pallette2:SetSize( 80, 60 )
pallette2:Reset()

local pallette3 = vgui.Create( &quot;DColorPalette&quot;, frame ) -- Third DColorPalette with 16 colors per row
pallette3:SetPos( 4, 200 )
pallette3:SetNumRows( 16 )
pallette3:SetSize( 160, 60 )
pallette3:Reset()

local pallette4 = vgui.Create( &quot;DColorPalette&quot;, frame ) -- Fourth DColorPalette with 24 colors per row
pallette4:SetPos( 4, 270 )
pallette4:SetNumRows( 24 )
pallette4:SetSize( 240, 60 )
pallette4:Reset()</code></pre>
