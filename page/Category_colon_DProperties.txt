<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a DProperties control with a few properties set up
</p>
<pre><code data-language="lua">local f = vgui.Create( &quot;DFrame&quot; )
f:SetSize( 500, 300 )
f:Center()
f:MakePopup()

local DProperties = vgui.Create( &quot;DProperties&quot;, f )
DProperties:Dock( FILL )

local Row1 = DProperties:CreateRow( &quot;Category1&quot;, &quot;Vector Color&quot; )
Row1:Setup( &quot;VectorColor&quot; )
Row1:SetValue( Vector( 1, 0, 0 ) )
Row1.DataChanged = function( _, val ) print( val ) end

local Row2 = DProperties:CreateRow( &quot;Category1&quot;, &quot;Combo&quot; )
Row2:Setup( &quot;Combo&quot;, { text = &quot;Select type...&quot; } )
Row2:AddChoice( &quot;Table&quot;, {} )
Row2:AddChoice( &quot;String&quot;, &quot;Hello world&quot; )
Row2.DataChanged = function( self, data )
	print( &quot;You selected: &quot;, data )
end

local Row3 = DProperties:CreateRow( &quot;Category1&quot;, &quot;Boolean&quot; )
Row3:Setup( &quot;Boolean&quot; )
Row3:SetValue( true )

local Row4 = DProperties:CreateRow( &quot;Category2&quot;, &quot;Float&quot; )
Row4:Setup( &quot;Float&quot;, { min = 0, max = 5 } )
Row4:SetValue( 2.5 )

local Row5 = DProperties:CreateRow( &quot;Category2&quot;, &quot;Integer&quot; )
Row5:Setup( &quot;Int&quot;, { min = 0, max = 5 } )
Row5:SetValue( 2.5 )

local Row6 = DProperties:CreateRow( &quot;Category2&quot;, &quot;Generic&quot; )
Row6:Setup( &quot;Generic&quot; )
Row6:SetValue( &quot;Hello World!&quot; )
Row6.DataChanged = function( _, val ) print( val ) end</code></pre>
<p><b>Output:</b>
</p>
<div style="margin:2em;">When you change the "Hello World!" textbox, it will print its new contents to the console.</div>
