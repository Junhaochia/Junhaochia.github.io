<h1> <span class="mw-headline" id="Description">Description</span></h1>
<p>Adds an option to the right side of the spawnmenu
</p>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_string" title="Category:string">string</a> tab</span>
</p>
<div style="margin-left: 32px;">
The spawnmenu tab to add into (for example "Utilities")</div>
</div>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments_2">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_string" title="Category:string">string</a> category</span>
</p>
<div style="margin-left: 32px;">
The category to add into (for example "Admin")</div>
</div>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments_3">Arguments</span></h1>

</p>

<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Adds a new option to the menu with a slider to change the gravity
</p>
<pre><code data-language="lua">hook.Add( &quot;AddToolMenuCategories&quot;, &quot;CustomCategory&quot;, function()
	spawnmenu.AddToolCategory( &quot;Utilities&quot;, &quot;Stuff&quot;, &quot;#Stuff&quot; )
end )

hook.Add( &quot;PopulateToolMenu&quot;, &quot;CustomMenuSettings&quot;, function()
	spawnmenu.AddToolMenuOption( &quot;Utilities&quot;, &quot;Stuff&quot;, &quot;Custom_Menu&quot;, &quot;#My Custom Menu&quot;, &quot;&quot;, &quot;&quot;, function( panel )
		panel:ClearControls()
		panel:NumSlider( &quot;Gravity&quot;, &quot;sv_gravity&quot;, 0, 600 )
		-- Add stuff here
	end )
end )</code></pre>
