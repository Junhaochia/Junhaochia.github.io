<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a bubble container panel in the middle of the screen. The background <a href="/page/Category_colon_DPanel" title="Category:DPanel">DPanel</a> is used to allow better control of the dialog point position.
</p>
<pre><code data-language="lua">--Background panel
BGPanel = vgui.Create(&quot;DPanel&quot;)
BGPanel:SetSize(300, 170)
BGPanel:Center()
BGPanel:SetDrawBackground(false)

-- The bubble container
local DBubbleContainer = vgui.Create(&quot;DBubbleContainer&quot;, BGPanel)

-- Set the dialog point 50 units from the left and make the entire bubble 280x150 pixels.
DBubbleContainer:OpenForPos(50, 0, 280, 150)</code></pre>
<p><br />

<h1 class="first_example"> <span class="mw-headline" id="Examples_2">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example_2">Example</span></h2>
<p>Creates a panel with Bill from Left 4 Dead talking through a speech bubble.
</p>
<pre><code data-language="lua">-- Image of Bill
local dimg = vgui.Create(&quot;DImage&quot;)
dimg:SetPos(0, ScrH()-512)
dimg:SetImage(&quot;spawnicons/models/survivors/survivor_namvet_128.png&quot;)
dimg:SizeToContents()

-- Determine placement of bubble container based on image dimensions
local dimg_x, dimg_y = dimg:GetPos()
local dimg_w, dimg_h = dimg:GetSize()

dimg_x = dimg_x + dimg_w/2
dimg_y = dimg_y + dimg_h/2.5

-- Bubble container size
local bubble_w, bubble_h = 150, 100

-- Create bubble container
local bubble = vgui.Create(&quot;DBubbleContainer&quot;)

-- Open it over Bill image
bubble:OpenForPos(dimg_x, dimg_y+bubble_h-32, bubble_w, bubble_h)

-- Add text to bubble
local lbl = vgui.Create(&quot;DLabel&quot;, bubble)
lbl:SetPos(5, 5)
lbl:SetSize(150, 50)
lbl:SetWrap(true)
lbl:SetFont(&quot;GModNotify&quot;)
lbl:SetText(&quot;Does this example get the point across?&quot;)
lbl:SetDark(true)</code></pre>
