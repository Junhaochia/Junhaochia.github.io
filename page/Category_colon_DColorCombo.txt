<h1> <span class="mw-headline" id="DescriptionPanelPage">Description<span style="display: none;"><a href="/page/Form:PanelPage" title="Form:PanelPage">PanelPage</a></span></span></h1>

</p>

<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a DColorCombo and sets its initial value.
</p>
<pre><code data-language="lua">local frame = vgui.Create( &quot;DFrame&quot; )
frame:SetSize( 500, 300 )
frame:Center()
frame:MakePopup()

local DermaColorCombo = vgui.Create( &quot;DColorCombo&quot;, frame )
DermaColorCombo:SetPos( 5, 30 )
DermaColorCombo:SetColor( Color( 255, 255, 255 ) )

local DColorButton = frame:Add( &quot;DColorButton&quot; )
DColorButton:SetPos( 300, 50 )
DColorButton:SetSize( 64, 64 )
function DermaColorCombo:OnValueChanged( col )
	DColorButton:SetColor( col )
end</code></pre>
