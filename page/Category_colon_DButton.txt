<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>The DButton is exactly what you think it is - a button!
</p>
<pre><code data-language="lua">local frame = vgui.Create( &quot;DFrame&quot; )
frame:SetSize( 300, 250 )
frame:Center()
frame:MakePopup()

local DermaButton = vgui.Create( &quot;DButton&quot;, frame ) // Create the button and parent it to the frame
DermaButton:SetText( &quot;Say hi&quot; )					// Set the text on the button
DermaButton:SetPos( 25, 50 )					// Set the position on the frame
DermaButton:SetSize( 250, 30 )					// Set the size
DermaButton.DoClick = function()				// A custom function run when clicked ( note the . instead of&#160;: )
	RunConsoleCommand( &quot;say&quot;, &quot;Hi&quot; )			// Run the console command &quot;say hi&quot; when you click it ( command, args )
end</code></pre>
