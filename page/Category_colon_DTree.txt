<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Example of using DTree
</p>
<pre><code data-language="lua">local frame = vgui.Create( &quot;DFrame&quot; )
frame:SetSize( 300, 500 )
frame:Center()
frame:MakePopup()

local dtree  = vgui.Create( &quot;DTree&quot;, frame )
dtree:Dock( FILL )

local node = dtree:AddNode( &quot;Node One&quot; )
local node = dtree:AddNode( &quot;Node Two&quot; )
local cnode = node:AddNode( &quot;Node 2.1&quot; )
local cnode = node:AddNode( &quot;Node 2.2&quot; )
local cnode = node:AddNode( &quot;Node 2.3&quot; )
local cnode = node:AddNode( &quot;Node 2.4&quot; )
local cnode = node:AddNode( &quot;Node 2.5&quot; )
local gcnode = cnode:AddNode( &quot;Node 2.5&quot; )
local cnode = node:AddNode( &quot;Node 2.6&quot; )
local node = dtree:AddNode( &quot;Node Three ( Maps Folder )&quot; )
node:MakeFolder( &quot;maps&quot;, &quot;GAME&quot;, true )
local node = dtree:AddNode( &quot;Node Four&quot; )</code></pre>
