<h1> <span class="mw-headline" id="DescriptionPanelPage">Description<span style="display: none;"><a href="/page/Form:PanelPage" title="Form:PanelPage">PanelPage</a></span></span></h1>
<p>A base for other controls for <a href="/page/Category:DProperties" title="Category:DProperties">DProperties</a>. Acts as a generic text input on its own.

<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Example usage of this control
</p>
<pre>local Panel = vgui.Create( &quot;DFrame&quot; )
Panel:SetSize( 500, 500 )
Panel:MakePopup()

local props = vgui.Create( &quot;DProperties&quot;, Panel )
props:Dock( FILL )

local textRow = props:CreateRow( &quot;MyCategory&quot;, &quot;GenericTextArea&quot; )
textRow:Setup( &quot;Generic&quot; )
textRow:SetValue( &quot;test&quot; )
textRow.DataChanged = function( self, data )
	MsgN( &quot;Text changed: &quot;, data )
end</pre>
