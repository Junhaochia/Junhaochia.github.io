<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Example of how you'd use this panel
</p>
<pre><code data-language="lua">local frame = vgui.Create( &quot;DFrame&quot; )
frame:SetSize( 500, 300 )
frame:Center()
frame:MakePopup()

local sheet = vgui.Create( &quot;DColumnSheet&quot;, frame )
sheet:Dock( FILL )

local panel1 = vgui.Create( &quot;DPanel&quot;, sheet )
panel1:Dock( FILL )
panel1.Paint = function( self, w, h ) draw.RoundedBox( 4, 0, 0, w, h, Color( 0, 128, 255 ) ) end 
sheet:AddSheet( &quot;test&quot;, panel1, &quot;icon16/cross.png&quot; )

local panel2 = vgui.Create( &quot;DPanel&quot;, sheet )
panel2:Dock( FILL )
panel2.Paint = function( self, w, h ) draw.RoundedBox( 4, 0, 0, w, h, Color( 255, 128, 0 ) ) end 
sheet:AddSheet( &quot;test 2&quot;, panel2, &quot;icon16/tick.png&quot; )</code></pre>
