<h1> <span class="mw-headline" id="DescriptionPanelPage">Description<span style="display: none;"><a href="/page/Form:PanelPage" title="Form:PanelPage">PanelPage</a></span></span></h1>
<p>A host for multiple <a href="/page/Category_colon_DCollapsibleCategory" title="Category:DCollapsibleCategory">DCollapsibleCategory</a> panels. As the name suggests, each one of them can be collapsed or expanded on demand by the user.
</p>

<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a DCategoryList with a category and a DFrame, as seen above.
</p>
<pre><code data-language="lua">local frame = vgui.Create( &quot;DFrame&quot; )
frame:SetSize( 500, 500 )
frame:Center()
frame:MakePopup()

local CatList = vgui.Create( &quot;DCategoryList&quot;, frame )
CatList:Dock( FILL )

local Cat = CatList:Add( &quot;Test category with text contents&quot; )
Cat:Add( &quot;Item 1&quot; )
local button = Cat:Add( &quot;Item 2&quot; )
button.DoClick = function()
	print( &quot;Item 2 was clicked.&quot; )
end

-- The contents can be any panel, even a DPanelList
local Cat2 = CatList:Add( &quot;Test category with panel contents&quot; )
Cat2:SetTall( 100 )

local Contents = vgui.Create( &quot;DButton&quot; )
Contents:SetText( &quot;This is the content of the category&quot; )
Cat2:SetContents( Contents )

CatList:InvalidateLayout( true )</code></pre>
