<h1> <span class="mw-headline" id="DescriptionPanelFunc">Description<span style="display: none;"><a href="/page/Form:PanelFunc" title="Form:PanelFunc">PanelFunc</a></span></span></h1>
<p>Sets the angles of the camera.
</p>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_Angle" title="Category:Angle">Angle</a> ang</span>
</p>
<div style="margin-left: 32px;">
The angles to set the camera to.</div>
</div>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a model panel with Eli rotating normally and the camera angles rotating sideways.
</p>
<pre><code data-language="lua">BGPanel = vgui.Create(&quot;DPanel&quot;)
BGPanel:SetPos(20, 20)
BGPanel:SetSize(400, 400)
BGPanel:SetBackgroundColor(Color(0, 0, 0, 255))

local mdl = vgui.Create(&quot;DModelPanel&quot;, BGPanel)
mdl:SetSize(BGPanel:GetSize())
mdl:SetModel(&quot;models/eli.mdl&quot;)

-- Position camera
mdl:SetCamPos(Vector(0, 60, 36))

local yaw = 0

function mdl:LayoutEntity(ent)

	-- Point camera toward the look pos
	local lookAng = (self.vLookatPos-self.vCamPos):Angle()
	
	-- Rotate the look angles based on incrementing yaw value
	lookAng:RotateAroundAxis(Vector(0, 1, 0), yaw)
	
	-- Set camera look angles
	self:SetLookAng(lookAng)
	
	-- Make entity rotate like normal
	ent:SetAngles(Angle(0, RealTime()*30,  0))
	
	yaw = yaw + 1

end</code></pre>
