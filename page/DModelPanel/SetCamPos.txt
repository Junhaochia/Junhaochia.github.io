<h1> <span class="mw-headline" id="DescriptionPanelFunc">Description<span style="display: none;"><a href="/page/Form:PanelFunc" title="Form:PanelFunc">PanelFunc</a></span></span></h1>
<p>Sets the position of the camera.
</p>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_Vector" title="Category:Vector">Vector</a> pos</span>
</p>
<div style="margin-left: 32px;">
The position to set the camera at.</div>
</div>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a model panel focused on Gman's face while he adjusts his tie.
</p>
<pre><code data-language="lua">BGPanel = vgui.Create(&quot;DPanel&quot;)
BGPanel:SetPos(20, 20)
BGPanel:SetSize(200, 200)
BGPanel:SetBackgroundColor(Color(0, 0, 0, 255))
	
local mdl = vgui.Create(&quot;DModelPanel&quot;, BGPanel)
mdl:SetSize(BGPanel:GetSize())
mdl:SetModel(&quot;models/player/gman_high.mdl&quot;)

function mdl:LayoutEntity(ent)
	ent:SetSequence(ent:LookupSequence(&quot;menu_gman&quot;))
	mdl:RunAnimation()		
	return
end

local eyepos = mdl.Entity:GetBonePosition(mdl.Entity:LookupBone(&quot;ValveBiped.Bip01_Head1&quot;))

eyepos:Add(Vector(0, 0, 2))	-- Move up slightly

mdl:SetLookAt(eyepos)

mdl:SetCamPos(eyepos-Vector(-12, 0, 0))	-- Move cam in front of eyes

mdl.Entity:SetEyeTarget(eyepos-Vector(-12, 0, 0))</code></pre>

<h1 class="first_example"> <span class="mw-headline" id="Examples_2">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example_2">Example</span></h2>
<p>Sets a model panel's camera position so the model won't go outside it
</p>
<pre><code data-language="lua">local mdlpnl = vgui.Create( &quot;DModelPanel&quot; )

local mn, mx = mdlpnl.Entity:GetRenderBounds()
local size = 0
size = math.max( size, math.abs(mn.x) + math.abs(mx.x) )
size = math.max( size, math.abs(mn.y) + math.abs(mx.y) )
size = math.max( size, math.abs(mn.z) + math.abs(mx.z) )

mdlpnl:SetFOV( 45 )
mdlpnl:SetCamPos( Vector( size, size, size ) )
mdlpnl:SetLookAt( (mn + mx) * 0.5 )</code></pre>
