<h1> <span class="mw-headline" id="Description">Description</span></h1>
<p>Causes a <a href="/page/Category_colon_RichText" title="Category:RichText">RichText</a> element to scroll to the top of its text.
</p>
<table class="gmodwiki_note bug_note">
<tr>
<td>BUG
</td>
<td>
<p>This does not work on the same frame as <a href="/page/Category_colon_Panel" title="Category:Panel">Panel</a>:<a href="/page/Panel/SetText" title="Panel/SetText">SetText</a>.
</p><p><b>Issue Tracker: <a rel="nofollow" class="external text" href="https://github.com/Facepunch/garrysmod-issues/issues/2239">#2239</a></b>
</p>
</td></tr></table>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a RichText panel with a "Back to Top" button which scrolls the text to the start.
</p>
<pre><code data-language="lua">-- Create a window frame
TextFrame = vgui.Create(&quot;DFrame&quot;)
TextFrame:SetSize(250, 200)
TextFrame:Center()
TextFrame:SetTitle(&quot;#ServerBrowser_ServerWarningTitle&quot;)
TextFrame:MakePopup()

-- RichText panel
local richtext = vgui.Create(&quot;RichText&quot;, TextFrame)
richtext:Dock(FILL)

-- Yellow colored localized text about player capacity
richtext:InsertColorChange(255, 255, 192, 255)
richtext:AppendText(&quot;#ServerBrowser_ServerWarning_MaxPlayers&quot;)

-- Create a button that moves the text back to the start
local topbutton = vgui.Create(&quot;DButton&quot;, richtext)
topbutton:SetSize(60, 20)
topbutton:SetPos(160, 146)
topbutton:SetText(&quot;Back to Top&quot;)

-- When clicked, go to the start of the text
topbutton.DoClick = function() richtext:GotoTextStart()	end

-- Apply background color and font
function richtext:PerformLayout()
	
	self:SetFontInternal(&quot;Trebuchet18&quot;)
	self:SetBGColor(Color(64, 64, 84))
	
end</code></pre>
