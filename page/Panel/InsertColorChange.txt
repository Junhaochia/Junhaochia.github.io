<h1> <span class="mw-headline" id="Description">Description</span></h1>
<p>Inserts a color change in a <a href="/page/Category_colon_RichText" title="Category:RichText">RichText</a> element, which affects the color of all text added with <a href="/page/Category_colon_Panel" title="Category:Panel">Panel</a>:<a href="/page/Panel/AppendText" title="Panel/AppendText">AppendText</a> until another color change is applied.
</p>

<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_number" title="Category:number">number</a> r</span>
</p>
<div style="margin-left: 32px;">
The red value (0 - 255).</div>
</div>

<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a RichText panel with color coding on certain segments of text.
</p>
<pre><code data-language="lua">-- Create a window frame
TextFrame = vgui.Create(&quot;DFrame&quot;)
TextFrame:SetSize(200, 200)
TextFrame:Center()
TextFrame:SetTitle(&quot;Colored text&quot;)
TextFrame:MakePopup()

-- RichText panel
local richtext = vgui.Create(&quot;RichText&quot;, TextFrame)
richtext:Dock(FILL)
richtext:SetVerticalScrollbarEnabled(false)

-- Text blocks
richtext:InsertColorChange(255, 255, 192, 255)
richtext:AppendText(&quot;This is an example of &quot;)

richtext:InsertColorChange(0, 255, 0, 255)
richtext:AppendText(&quot;color coding &quot;)

richtext:InsertColorChange(255, 255, 192, 255)
richtext:AppendText(&quot;different segments of text throughout a &quot;)

richtext:InsertColorChange(255, 200, 0, 255)
richtext:AppendText(&quot;Rich Text panel.\n\n&quot;)

richtext:InsertColorChange(64, 0, 255, 255)
richtext:AppendText(&quot;Here is another line of text shown in the color &quot;)

richtext:InsertColorChange(128, 0, 255, 255)
richtext:AppendText(&quot;purple.&quot;)

-- When the panel is ready for layout, apply font and background color
function richtext:PerformLayout()
	
	self:SetFontInternal(&quot;Trebuchet18&quot;)
	self:SetBGColor(Color(0, 16, 32))
	
end</code></pre>

<h1 class="first_example"> <span class="mw-headline" id="Examples_2">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example_2">Example</span></h2>
<p>Word by word coloring using <a href="/page/Category_colon_string" title="Category:string">string</a>.<a href="/page/string/Explode" title="string/Explode">Explode</a> and random colors.
</p>
<pre><code data-language="lua">-- Create a window frame
TextFrame = vgui.Create(&quot;DFrame&quot;)
TextFrame:SetSize(300, 200)
TextFrame:Center()
TextFrame:SetTitle(&quot;Randomly Colored Words&quot;)
TextFrame:MakePopup()

-- RichText panel
local richtext = vgui.Create(&quot;RichText&quot;, TextFrame)
richtext:Dock(FILL)
richtext:SetVerticalScrollbarEnabled(false)

local txt = &quot;Here's a fun example involving word by word text coloring. &quot;..
&quot;Each word is separated by a space, colored, and appended to the Rich Text panel individually. &quot;..
&quot;The colors are randomly generated shades of red, orange, yellow, and pink.&quot;

local txt_tbl = string.Explode(&quot; &quot;, txt, false)

for _, word in pairs(txt_tbl) do

	richtext:InsertColorChange(255, math.random(0, 255), math.random(0, 255), 255)
	richtext:AppendText(word..&quot; &quot;)

end

function richtext:PerformLayout()
	
	self:SetFontInternal(&quot;GModNotify&quot;)
	self:SetBGColor(Color(32, 16, 0))
	
end</code></pre>
