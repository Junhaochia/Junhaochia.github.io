<h1> <span class="mw-headline" id="Description">Description</span></h1>
<p>Sets the drag parent.
</p><p>Drag parent means that when we start to drag this panel, we'll really start dragging the defined parent.
</p>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_Panel" title="Category:Panel">Panel</a> parent</span>
</p>
<div style="margin-left: 32px;">
The panel to set as drag parent.</div>
</div>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Example usage.
</p>
<pre><code data-language="lua">local function DoDrop( self, panels, bDoDrop, Command, x, y )
	if ( bDoDrop ) then
		for k, v in pairs( panels ) do
			self:AddItem( v )
		end
	end
end

concommand.Add( &quot;test2&quot;, function()

	local frame = vgui.Create( &quot;DFrame&quot; )
	frame:SetSize( 500, 300 )
	frame:SetTitle( &quot;Frame&quot; )
	frame:MakePopup()
	frame:Center()

	local left = vgui.Create( &quot;DScrollPanel&quot;, frame )
	left:Dock( LEFT )
	left:SetWidth( frame:GetWide() / 2 - 7 )
	left:SetPaintBackground( true )
	left:DockMargin( 0, 0, 4, 0 )
	left:Receiver( &quot;myDNDname&quot;, DoDrop ) -- Make the panel a receiver for drag and drop events

	local right = vgui.Create( &quot;DScrollPanel&quot;, frame )
	right:Dock( FILL )
	right:SetPaintBackground( true )
	right:Receiver( &quot;myDNDname&quot;, DoDrop )

	for i = 1, 30 do
		local but = vgui.Create( &quot;DButton&quot; )
		but:SetText( &quot;button text &quot; .. i )
		but:SetSize( 36, 24 )
		but:Dock( TOP )
		but:Droppable( &quot;myDNDname&quot; ) -- make the panel be able to be drag'n'dropped onto other panels
		right:AddItem( but )
		
		local mdl = vgui.Create( &quot;DModelPanel&quot;, but ) -- Put another panel over the draggable button
		mdl:Dock( FILL )
		mdl:SetModel( &quot;models/alyx.mdl&quot; )
		mdl:SetDragParent( but ) -- The magic bit. Without it, you would not be able to drag the button at all
	end

end )</code></pre>
