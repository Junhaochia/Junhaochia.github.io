<h1> <span class="mw-headline" id="Description">Description</span></h1>
<p>Sends an action command signal to the panel. The response is handled by <a href="/page/Category_colon_PANEL_Hooks" title="Category:PANEL Hooks">PANEL</a>:<a href="/page/PANEL/ActionSignal" title="PANEL/ActionSignal">ActionSignal</a>.
</p>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_string" title="Category:string">string</a> command</span>
</p>
<div style="margin-left: 32px;">
The command to send to the panel.</div>
</div>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Defines a function which displays a dialog box that asks the player if they want to leave the server. Clicking the buttons fire panel commands that call the <a href="/page/Category_colon_PANEL_Hooks" title="Category:PANEL Hooks">PANEL</a>:<a href="/page/PANEL/ActionSignal" title="PANEL/ActionSignal">ActionSignal</a> which handles what action should be taken.
</p>
<pre><code data-language="lua">function DisconnectDialog()

	-- Remove existing dialog box
	if(DialogBox) then DialogBox:Remove() end
	
	-- Font and message
	local font = &quot;ChatFont&quot;
	local msg = &quot;Would you like to disconnect?&quot;
	
	-- Get the size of the text
	surface.SetFont(font)
	local msg_w, msg_h = surface.GetTextSize(msg)
	
	-- Padding and button size
	local padding = 15
	local btn_w, btn_h = 35, 25
	
	-- Calculate dialog box size
	local dialog_w = msg_w+(padding*2)
	local dialog_h = msg_h+(padding*3)+btn_h
	
	-- Create the dialog box
	DialogBox = vgui.Create(&quot;DPanel&quot;)
	DialogBox:SetSize(dialog_w, dialog_h)
	DialogBox:Center()
	DialogBox:SetBackgroundColor(Color(64, 64, 92, 255))
	
	-- Message
	local lbl = vgui.Create(&quot;DLabel&quot;, DialogBox)
	lbl:SetPos(padding, padding)
	lbl:SetSize(msg_w, msg_h)
	lbl:SetText(msg)
	lbl:SetFont(font)
	
	-- Yes button
	local yes = vgui.Create(&quot;DButton&quot;, DialogBox)
	yes:SetPos((dialog_w/2)-btn_w-20, msg_h+padding*2)
	yes:SetSize(btn_w, btn_h)
	yes:SetText(&quot;Yes&quot;)
	yes:SetFont(font)
	yes.DoClick = function() DialogBox:Command(&quot;yes&quot;) end
	
	-- No button
	local no = vgui.Create(&quot;DButton&quot;, DialogBox)
	no:SetPos((dialog_w/2)+20, msg_h+padding*2)
	no:SetSize(btn_w, btn_h)
	no:SetText(&quot;No&quot;)
	no:SetFont(font)
	no.DoClick = function() DialogBox:Command(&quot;no&quot;) end
	
	-- Force mouse input
	DialogBox:MakePopup()
	
	-- Hook fired when DialogBox:Command is used
	function DialogBox:ActionSignal(signalName, signalValue)
	
		-- Thank the player and disconnect after 2 seconds
		if(signalName == &quot;yes&quot;) then
		
			chat.AddText(Color(192, 192, 224), &quot;Thanks for playing &quot;..LocalPlayer():Nick()..&quot;, come back soon!&quot;)
			
			timer.Simple(2.0, function()
				RunConsoleCommand(&quot;disconnect&quot;)
			end)
			
			self:Remove()
		
		-- Remove the dialog box
		elseif(signalName == &quot;no&quot;) then
		
			self:Remove()
		
		end
	
	end
	
end</code></pre>
