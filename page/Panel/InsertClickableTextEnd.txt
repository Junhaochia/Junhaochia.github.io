<h1> <span class="mw-headline" id="Description">Description</span></h1>
<p>Marks the end of a clickable text segment in a <a href="/page/Category_colon_RichText" title="Category:RichText">RichText</a> element, started with <a href="/page/Category_colon_Panel" title="Category:Panel">Panel</a>:<a href="/page/Panel/InsertClickableTextStart" title="Panel/InsertClickableTextStart">InsertClickableTextStart</a>.
</p>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a panel with some information on RichText panels along with a click-able link to the <a href="/page/Category_colon_RichText" title="Category:RichText">RichText</a> page.
</p>
<pre><code data-language="lua">-- Create a window frame
TextFrame = vgui.Create(&quot;DFrame&quot;)
TextFrame:SetSize(250, 150)
TextFrame:Center()
TextFrame:SetTitle(&quot;RichText&quot;)
TextFrame:MakePopup()

-- RichText panel
local richtext = vgui.Create(&quot;RichText&quot;, TextFrame)
richtext:Dock(FILL)

-- First segment
richtext:InsertColorChange(255, 255, 255, 255)
richtext:AppendText(&quot;This is a Rich Text panel — a panel used in Source MP's default chat box and developer console.\n\nSee the &quot;)

-- Second segment
richtext:InsertColorChange(192, 192, 255, 255)
richtext:InsertClickableTextStart(&quot;OpenWiki&quot;)	-- Make incoming text fire the &quot;OpenWiki&quot; value when clicked
richtext:AppendText(&quot;Garry's Mod Wiki&quot;)
richtext:InsertClickableTextEnd()	-- End clickable text here

-- Third segment
richtext:InsertColorChange(255, 255, 255, 255)
richtext:AppendText(&quot; for information on how to use a Rich Text panel.&quot;)

-- Background color
function richtext:PerformLayout() self:SetBGColor(Color(32, 32, 46)) end

-- Handle any commands we get from the panel
function richtext:ActionSignal(signalName, signalValue)

	-- Some clickable text was clicked
	if(signalName == &quot;TextClicked&quot;) then
	
		-- Open the wiki
		if(signalValue == &quot;OpenWiki&quot;) then
		
			gui.OpenURL(&quot;http://wiki.garrysmod.com/page/Category_colon_RichText&quot;)
		
		end
	
	end

end</code></pre>
