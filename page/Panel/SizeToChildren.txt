<h1> <span class="mw-headline" id="Description">Description</span></h1>
<p>Resizes the panel to fit the bounds of its children.
</p>
<table cellspacing="0" class="gmodwiki_note">
<tr>
<td>NOTE
</td>
<td>
<p>Your panel must have its layout updated (<a href="/page/Category_colon_Panel" title="Category:Panel">Panel</a>:<a href="/page/Panel/InvalidateLayout" title="Panel/InvalidateLayout">InvalidateLayout</a>) for this function to work properly.
</p>
</td></tr></table>
<table cellspacing="0" class="gmodwiki_note">
<tr>
<td>NOTE
</td>
<td>
<p>The sizeW and sizeH parameters are false by default. Therefore, calling this function with no arguments will result in a no-op.
</p>
</td></tr></table>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_boolean" title="Category:boolean">boolean</a> sizeW=false</span>
</p>
<div style="margin-left: 32px;">
Resize with width of the panel.</div>
</div>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments_2">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_boolean" title="Category:boolean">boolean</a> sizeH=false</span>
</p>
<div style="margin-left: 32px;">
Resize the height of the panel.</div>
</div>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Using <a href="/page/Category_colon_Panel" title="Category:Panel">Panel</a>:<a href="/page/Panel/InvalidateLayout" title="Panel/InvalidateLayout">InvalidateLayout</a>
</p>
<pre><code data-language="lua">local Frame = vgui.Create( &quot;DFrame&quot; )
Frame:SetSize( 300, 400 )
Frame:Center()
Frame:MakePopup()
Frame:SetSizable( true )

-- with&#160;:InvalidateLayout(true)
local backgroundPanel = vgui.Create( &quot;DPanel&quot;, Frame )
backgroundPanel:Dock( TOP )
backgroundPanel:DockPadding( 4, 4, 4, 4 )
backgroundPanel:DockMargin( 0, 0, 0, 4 )

local button1 = vgui.Create( &quot;DButton&quot;, backgroundPanel )
button1:Dock( TOP )
button1:DockMargin( 0, 0, 0, 4 )
button1:SetTall( 60 )
button1:SetText( &quot;c1&quot; )

local button2 = vgui.Create( &quot;DButton&quot;, backgroundPanel )
button2:Dock( TOP )
button2:SetTall( 60 )
button2:SetText( &quot;c2&quot; )

backgroundPanel:InvalidateLayout( true )
backgroundPanel:SizeToChildren( false, true )


-- now w/o&#160;:InvalidateLayout
local backgroundPanel = vgui.Create( &quot;DPanel&quot;, Frame )
backgroundPanel:Dock( TOP )
backgroundPanel:DockPadding( 4, 4, 4, 4 )

local button1 = vgui.Create( &quot;DButton&quot;, backgroundPanel )
button1:Dock( TOP )
button1:DockMargin( 0, 0, 0, 4 )
button1:SetTall( 60 )
button1:SetText( &quot;c1&quot; )

local button2 = vgui.Create( &quot;DButton&quot;, backgroundPanel )
button2:Dock( TOP )
button2:SetTall( 60 )
button2:SetText( &quot;c2&quot; )

backgroundPanel:SizeToChildren( false, true )</code></pre>
