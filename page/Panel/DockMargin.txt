<h1> <span class="mw-headline" id="Description">Description</span></h1>
<p>Sets the dock margin of the panel.
</p><p>The dock margin is the extra space that will be left around the edge when this element is docked inside its parent element.
</p>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">

</p>
<div style="margin-left: 32px;">
The left margin to the parent.</div>
</div>

<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Example showing the effects of DockMargin and DockPadding
</p>
<pre><code data-language="lua">local frame = vgui.Create(&quot;DFrame&quot;)
frame:SetSize(600, 300)
frame:SetTitle(&quot;Docking Demonstration&quot;)
frame:Center()
frame:MakePopup(true)

local panel = vgui.Create(&quot;DPanel&quot;, frame)
panel:DockMargin(10, 20, 30, 40)
panel:DockPadding(40, 30, 20, 10)
panel:Dock(FILL)

local button = vgui.Create(&quot;DButton&quot;, panel)
button:SetText(&quot;Lopsided proportions!&quot;)
button:Dock(FILL)</code></pre>
