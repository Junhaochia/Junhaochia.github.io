<h1> <span class="mw-headline" id="Description">Description</span></h1>
<p>Sets the dock type of the panel.
</p>
<table cellspacing="0" class="gmodwiki_note">
<tr>
<td>NOTE
</td>
<td>
<p>After using this function, if you want to get the correct panel's bounds (position, size), use <a href="/page/Category_colon_Panel" title="Category:Panel">Panel</a>:<a href="/page/Panel/InvalidateParent" title="Panel/InvalidateParent">InvalidateParent</a> (use <b>true</b> as argument if you need to update immediately)
</p>
</td></tr></table>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_number" title="Category:number">number</a> dockType</span>
</p>
<div style="margin-left: 32px;">
Dock type using <a href="/page/Enums/DOCK" title="Enums/DOCK">DOCK_ Enums</a>.</div>
</div>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Example docking including DockMargin. Provided by Walrus Viking in <a rel="nofollow" class="external text" href="http://facepunch.com/showthread.php?t=1439021&amp;p=47095061&amp;viewfull=1#post47095061">this</a> Facepunch post.
</p>
<pre><code data-language="lua">local f = vgui.Create( &quot;DFrame&quot; )
f:SetTitle( &quot;Dock Test&quot; )
f:SetSize( 256, 256 )
f:Center()
f:MakePopup()

local p = vgui.Create( &quot;DPanel&quot;, f )
p:Dock( FILL )
p:DockMargin( 0, 0, 0, 0 )

for i = 0, 10, 1 do
	local l = vgui.Create( &quot;DLabel&quot;, p )
	l:Dock( TOP )
	l:DockMargin( 4, 0, 0, 0 ) -- shift to the right
	l:SetColor( color_black )
	l:SetText( &quot;Hi! I'm a label!&quot; )
end</code></pre>

<h1 class="first_example"> <span class="mw-headline" id="Examples_2">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example_2">Example</span></h2>
<p>Example showing how multiple docked elements behave.
</p>
<pre><code data-language="lua">local frame = vgui.Create(&quot;DFrame&quot;)
frame:SetSize(600, 300)
frame:SetTitle(&quot;Docking Demonstration&quot;)
frame:Center()
frame:MakePopup(true)

local panel = vgui.Create(&quot;DPanel&quot;, frame) --Create a panel on the left
panel:SetSize(300, 0) --Height doesn't matter since we're docking it to the left anyways
panel:Dock(LEFT)
local fill = vgui.Create(&quot;DButton&quot;, panel) --Create a button and dock it
fill:SetText(&quot;FILL&quot;)
fill:Dock(FILL)
local left = vgui.Create(&quot;DButton&quot;, panel)
left:SetText(&quot;LEFT&quot;)
left:Dock(LEFT)
local right = vgui.Create(&quot;DButton&quot;, panel)
right:SetText(&quot;RIGHT&quot;)
right:Dock(RIGHT)
local top = vgui.Create(&quot;DButton&quot;, panel)
top:SetText(&quot;TOP&quot;)
top:Dock(TOP)
local bottom = vgui.Create(&quot;DButton&quot;, panel)
bottom:SetText(&quot;BOTTOM&quot;)
bottom:Dock(BOTTOM)

local panel = vgui.Create(&quot;DPanel&quot;, frame) --Do the same thing on the right, but this time with top and bottom before left and right
panel:SetSize(300, 0)
panel:Dock(RIGHT)
local fill = vgui.Create(&quot;DButton&quot;, panel)
fill:SetText(&quot;FILL&quot;)
fill:Dock(FILL)
local top = vgui.Create(&quot;DButton&quot;, panel)
top:SetText(&quot;TOP&quot;)
top:Dock(TOP)
local bottom = vgui.Create(&quot;DButton&quot;, panel)
bottom:SetText(&quot;BOTTOM&quot;)
bottom:Dock(BOTTOM)
local left = vgui.Create(&quot;DButton&quot;, panel)
left:SetText(&quot;LEFT&quot;)
left:Dock(LEFT)
local right = vgui.Create(&quot;DButton&quot;, panel)
right:SetText(&quot;RIGHT&quot;)
right:Dock(RIGHT)</code></pre>
