<h1> <span class="mw-headline" id="Description">Description</span></h1>
<p>When TAB is pressed, the next selectable panel in the number sequence is selected.
</p>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_number" title="Category:number">number</a> position</span>
</p>
</div>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a menu to put information in with SetTablePosition application.
</p>
<pre><code data-language="lua">local Panel = vgui.Create(&quot;DFrame&quot;)
Panel:SetSize(500,250)
Panel:Center()
Panel:SetText(&quot;My Information&quot;)
Panel:MakePopup()

local FirstName = vgui.Create(&quot;DTextEntry&quot;, Panel)
FirstName:SetSize(400,35)
FirstName:SetPos(50, 50)
FirstName:SetPlaceholderText(&quot;First Name&quot;)
FirstName:SetTabPosition( 1 )

local LastName = vgui.Create(&quot;DTextEntry&quot;, Panel)
LastName:SetSize(400,35)
LastName:SetPos(50, 100)
LastName:SetPlaceholderText(&quot;Last Name&quot;)
LastName:SetTabPosition( 2 )

local FavoriteColor = vgui.Create(&quot;DTextEntry&quot;, Panel)
FavoriteColor:SetSize(400,35)
FavoriteColor:SetPos(50, 150)
FavoriteColor:SetPlaceholderText(&quot;Favorite Color&quot;)
FavoriteColor:SetTabPosition( 3 )

local CompletedButton = vgui.Create(&quot;DButton&quot;, Panel)
CompletedButton:SetSize(200,35)
CompletedButton:SetPos(150, 200)
CompletedButton:SetText(&quot;Done&quot;)
function CompletedButton:DoClick()
    LocalPlayer():ConCommand(&quot;say My name is &quot; .. FirstName:GetText() .. &quot; &quot; .. LastName:GetText() .. &quot; and my favorite color is &quot; .. FavoriteColor:GetText() .. &quot;!&quot;)
    Panel:Remove()
end</code></pre>
