<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a DKillIcon and sets it to the crowbar killicon.
</p>
<pre><code data-language="lua">local frame = vgui.Create(&quot;DFrame&quot;)
frame:SetSize(200,200)
frame:Center()
frame:SetTitle(&quot;DKillIcon Example&quot;)

local icon = vgui.Create(&quot;DKillIcon&quot;, frame)
icon:SetName(&quot;weapon_crowbar&quot;)
icon:SizeToContents()
icon:Center()</code></pre>
