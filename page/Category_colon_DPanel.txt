<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a DPanel and adds a <a href="/page/Category_colon_DLabel" title="Category:DLabel">DLabel</a> to it
</p>
<pre><code data-language="lua">local DPanel = vgui.Create( &quot;DPanel&quot; )
DPanel:SetPos( 10, 30 ) -- Set the position of the panel
DPanel:SetSize( 200, 200 ) -- Set the size of the panel

local DLabel = vgui.Create( &quot;DLabel&quot;, DPanel )
DLabel:SetPos( 10, 10 ) -- Set the position of the label
DLabel:SetText( &quot;I'm a DLabel inside a DPanel!&#160;:)&quot; ) --  Set the text of the label
DLabel:SizeToContents() -- Size the label to fit the text in it
DLabel:SetDark( 1 ) -- Set the colour of the text inside the label to a darker one</code></pre>
