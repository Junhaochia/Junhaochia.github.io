<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>An example usage of the function
</p>
<pre><code data-language="lua">--This creates the frame.
local Frame = vgui.Create( &quot;DFrame&quot; )
Frame&#160;:Center() -- or Frame:SetPos( x, y )
Frame:SetSize( 300, 150 )
Frame:SetTitle( &quot;Test&quot; )
Frame:SetVisible( true )
Frame:SetDraggable( true )
Frame:ShowCloseButton( true )
Frame:MakePopup()
--Here we create the slider.
local DermaSlider = vgui.Create( &quot;DNumSlider&quot;, Frame )
DermaSlider:SetPos( 25, 85 )
DermaSlider:SetWide( 275 )
DermaSlider:SetMin( 0 ) -- Or 3 for second image
DermaSlider:SetMax( 5 )
DermaSlider:SetValue( 0.5 )
DermaSlider:SetDecimals( 2 )</code></pre>
