<h1> <span class="mw-headline" id="DescriptionPanelPage">Description<span style="display: none;"><a href="/page/Form:PanelPage" title="Form:PanelPage">PanelPage</a></span></span></h1>

<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates the DShape rectangle within a small popup frame
</p>
<pre><code data-language="lua">local Frame = vgui.Create( &quot;DFrame&quot; ) -- Create a frame 
Frame:SetTitle( &quot;Test panel&quot; )
Frame:SetSize(400,400)
Frame:Center()			
Frame:MakePopup()

local Shape = vgui.Create( &quot;DShape&quot;, Frame )
Shape:SetType( &quot;Rect&quot; ) -- This is the only type it can be
Shape:SetPos( 100, 100 ) 
Shape:SetColor( Color(0, 255, 0, 255) )
Shape:SetSize( 200, 200 )</code></pre>
