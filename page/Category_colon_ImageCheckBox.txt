<h1> <span class="mw-headline" id="DescriptionPanelPage">Description<span style="display: none;"><a href="/page/Form:PanelPage" title="Form:PanelPage">PanelPage</a></span></span></h1>
<p>A checkbox panel similar to <a href="/page/Category_colon_DCheckBox" title="Category:DCheckBox">DCheckBox</a> and <a href="/page/Category_colon_DImageButton" title="Category:DImageButton">DImageButton</a> with customizable checked state image.
</p><p>Uses the <a href="/page/Category_colon_Material" title="Category:Material">Material</a> panel internally. Can't have a label.
</p>
<table cellspacing="0" class="gmodwiki_note">
<tr>
<td>NOTE
</td>
<td>
<p>Will error if no material was set.
</p>
</td></tr></table>

<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Example of <strong class="selflink">ImageCheckBox</strong> usage.
</p>
<pre><code data-language="lua">local DFrame = vgui.Create( &quot;DFrame&quot; ) -- Create DFrame
DFrame:SetSize( 300, 305 ) -- Set its size
DFrame:Center() -- Center it
DFrame:SetTitle( &quot;Check list&quot; ) -- Set a title for it
DFrame:MakePopup() -- Make popup
DFrame.Items = {} -- Create table to store lines

local Head = DFrame:Add( &quot;DPanel&quot; ) -- Create head line
Head:Dock( TOP ) -- Dock it
Head:SetHeight( 20 ) -- Set height

function Head:Paint( w, h )
	local completed = 0 -- Variable to store amount of checked ImageCheckBox'es

	for k, v in ipairs( DFrame.Items ) do
		if ( v.ImageCheckBox:GetChecked() ) then
			completed = completed + 1 -- Get ImageCheckBox'es state and count checked
		end
	end

	draw.SimpleText( &quot;You've completed &quot; .. completed .. &quot; of &quot; .. #DFrame.Items .. &quot; items&quot;, &quot;DermaDefaultBold&quot;, w / 2, h / 2, Color( 255, 255, 255 ), TEXT_ALIGN_CENTER, TEXT_ALIGN_CENTER ) -- Draw the text
end

-- Convenience function to quickly add items
local function addItem( text )
	local RulePanel = DFrame:Add( &quot;DPanel&quot; ) -- Create container for this item
	RulePanel:Dock( TOP ) -- Dock it
	RulePanel:DockMargin( 0, 1, 0, 0 ) -- Add 1 pixel spacing in between each item
	table.insert( DFrame.Items, RulePanel ) -- Add to list of lines

	local ImageCheckBox = RulePanel:Add( &quot;ImageCheckBox&quot; ) -- Create checkbox with image
	ImageCheckBox:SetMaterial( &quot;icon16/accept.png&quot; ) -- Set its image
	ImageCheckBox:SetWidth( 24 ) -- Make the check box a bit wider than the image so it looks nicer
	ImageCheckBox:Dock( LEFT ) -- Dock it
	ImageCheckBox:SetChecked( false ) -- Set unchecked
	RulePanel.ImageCheckBox = ImageCheckBox -- Add reference to call

	local DLabel = RulePanel:Add( &quot;DLabel&quot; ) -- Create text
	DLabel:SetText( text ) -- Set the text
	DLabel:Dock( FILL ) -- Dock it
	DLabel:DockMargin( 5, 0, 0, 0 ) -- Move the text to the right a little
	DLabel:SetTextColor( Color( 0, 0, 0 ) ) -- Set text color to black
end

-- Adding items
addItem( &quot;Learn something&quot; )
addItem( &quot;Do something&quot; )
addItem( &quot;Make something&quot; )
addItem( &quot;Create something&quot; )
addItem( &quot;Play something&quot; )
addItem( &quot;Test something&quot; )
addItem( &quot;Write a really long item for testing purposes&quot; )
addItem( &quot;Break something&quot; )
addItem( &quot;Rebuild something&quot; )
addItem( &quot;Release something&quot; )</code></pre>
