<h1> <span class="mw-headline" id="DescriptionPanelPage">Description<span style="display: none;"><a href="/page/Form:PanelPage" title="Form:PanelPage">PanelPage</a></span></span></h1>

<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a DImage of Dr. Breen inside a <a href="/page/Category_colon_DFrame" title="Category:DFrame">DFrame</a> panel.
</p>
<pre><code data-language="lua">-- Frame
Frame = vgui.Create(&quot;DFrame&quot;)
Frame:SetSize(200, 200)
Frame:Center()
Frame:SetTitle(&quot;Image of Dr. Breen&quot;)	-- Title of window

-- Image panel of Dr. Breen
local breen_img = vgui.Create(&quot;DImage&quot;, Frame)	-- Add image to Frame
breen_img:SetPos(10, 35)	-- Move it into frame
breen_img:SetSize(150, 150)	-- Size it to 150x150

-- Set material relative to &quot;garrysmod/materials/&quot;
breen_img:SetImage(&quot;scripted/breen_fakemonitor_1&quot;)</code></pre>
<p><br />
</p>
<h1 class="first_example"> <span class="mw-headline" id="Examples_2">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example_2">Example</span></h2>
<p>Creates three DImage panels and arranges them into a new image.
</p>
<pre><code data-language="lua">-- Container panel
BGPanel = vgui.Create(&quot;DPanel&quot;)
BGPanel:SetSize(400, 400)
BGPanel:Center()
BGPanel:SetDrawBackground(false)

-- Wood background
local img_bg = vgui.Create(&quot;DImage&quot;, BGPanel)
img_bg:SetSize(BGPanel:GetSize())		
img_bg:SetImage(&quot;models/props_foliage/oak_tree01&quot;)

-- Blurred out screenshot of Construct
local img_construct = vgui.Create(&quot;DImage&quot;, BGPanel)
img_construct:SetPos(10, 10)
img_construct:SetSize(380, 380)		
img_construct:SetImage(&quot;hlmv/background&quot;)

-- Flatgrass sign
local img_text = vgui.Create(&quot;DImage&quot;, BGPanel)
img_text:SetPos(10, 20)
img_text:SetSize(380, 130)		
img_text:SetImage(&quot;gm_construct/flatsign&quot;)</code></pre>
