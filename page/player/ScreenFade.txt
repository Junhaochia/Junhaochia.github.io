</p>
<pre><code data-language="lua">hook.Add( &quot;PlayerHurt&quot;, &quot;hurt_effect_fade&quot;, function( ply )
	ply:ScreenFade( SCREENFADE.IN, Color( 255, 0, 0, 128 ), 0.3, 0 )
end )</code></pre>
