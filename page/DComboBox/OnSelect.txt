<h1> <span class="mw-headline" id="DescriptionPanelHook">Description<span style="display: none;"><a href="/page/Form:PanelHook" title="Form:PanelHook">PanelHook</a></span></span></h1>
<p>Called when an option in the combo box is selected.
</p>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_number" title="Category:number">number</a> index</span>
</p>
<div style="margin-left: 32px;">
The index of the option for use with other <a href="/page/Category_colon_DComboBox" title="Category:DComboBox">DComboBox</a> functions.</div>
</div>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments_2">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_string" title="Category:string">string</a> value</span>
</p>
<div style="margin-left: 32px;">
The name of the option.</div>
</div>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments_3">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_any" title="Category:any">any</a> data</span>
</p>
<div style="margin-left: 32px;">
The data assigned to the option.</div>
</div>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a combo box that controls the color of the background panel.
</p>
<pre><code data-language="lua">local frame = vgui.Create(&quot;DFrame&quot;)
frame:SetSize( 500, 500 )
frame:Center()
frame:MakePopup()

-- Background panel
local BGPanel = vgui.Create( &quot;DPanel&quot;, frame )
BGPanel:Dock( FILL )

local cbox = vgui.Create( &quot;DComboBox&quot;, BGPanel )
cbox:SetPos( 5, 5 )
cbox:SetSize( 190, 20 )
cbox:SetValue( &quot;Pick a color&quot; ) -- Default text

-- Color choices
cbox:AddChoice( &quot;Red&quot;, Color( 255, 0, 0 ) )
cbox:AddChoice( &quot;Orange&quot;, Color( 255, 128, 0 ) )
cbox:AddChoice( &quot;Yellow&quot;, Color( 255, 255, 0 ) )
cbox:AddChoice( &quot;Green&quot;, Color( 0, 255, 0 ) )
cbox:AddChoice( &quot;Blue&quot;, Color( 0, 0, 255 ) )
cbox:AddChoice( &quot;Indigo&quot;, Color( 64, 0, 255 ) )
cbox:AddChoice( &quot;Violet&quot;, Color( 128, 0, 255 ) )
cbox:AddChoice( &quot;Pink&quot;, Color( 255, 0, 255 ) )

function cbox:OnSelect( index, text, data )

	-- Set background panel color
	BGPanel:SetBackgroundColor( data )

end</code></pre>
