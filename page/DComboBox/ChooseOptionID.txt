<h1> <span class="mw-headline" id="DescriptionPanelFunc">Description<span style="display: none;"><a href="/page/Form:PanelFunc" title="Form:PanelFunc">PanelFunc</a></span></span></h1>
<p>Selects an option within a combo box based on its table index.
</p>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_number" title="Category:number">number</a> index</span>
</p>
<div style="margin-left: 32px;">
Selects the option with given index.</div>
</div>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>A simple combo box menu which gives choices for a favorite lunch meal, including a non-preference choice which randomly chooses an option.
</p>
<pre><code data-language="lua">BGPanel = vgui.Create(&quot;DPanel&quot;)
BGPanel:SetPos(20, 20)
BGPanel:SetSize(200, 100)

-- Text output
local lbl = vgui.Create(&quot;DLabel&quot;, BGPanel)
lbl:SetPos(10, 80)
lbl:SetSize(180, 20)
lbl:SetDark(true)
lbl:SetText(&quot;You choose...&quot;)

-- Combo box
local cbox = vgui.Create(&quot;DComboBox&quot;, BGPanel)
cbox:SetPos(5, 5)
cbox:SetSize(190, 20)

cbox:SetValue(&quot;What's your favorite lunch meal?&quot;)

-- Choices
cbox:AddChoice(&quot;BBQ Chicken&quot;)
cbox:AddChoice(&quot;Fish and Chips&quot;)
cbox:AddChoice(&quot;Pizza&quot;)
cbox:AddChoice(&quot;Potato Salad&quot;)
cbox:AddChoice(&quot;Roast Beef Sandwich&quot;)
cbox:AddChoice(&quot;Spaghetti&quot;)

-- No preference: data is set to -1
cbox:AddChoice(&quot;I don't have a favorite.&quot;, -1)

function cbox:OnSelect(index, value, data)

	-- No preference? Choose a random choice
	if(data == -1) then				
		self:ChooseOptionID(math.random(1, 6))
		
	-- Otherwise update the text label with our choice
	else
		lbl:SetText(&quot;You choose &quot;..value..&quot;.&quot;)
	end
	
end</code></pre>
