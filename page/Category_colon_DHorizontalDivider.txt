<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Insert a horizontal divider between two <a href="/page/Category_colon_DPanel" title="Category:DPanel">DPanels</a> in a <a href="/page/Category_colon_DFrame" title="Category:DFrame">DFrame</a>.
</p>
<pre><code data-language="lua">local f = vgui.Create( &quot;DFrame&quot; )
f:SetSize( 400, 200 )
f:Center()
f:MakePopup()

local LeftPanel = vgui.Create( &quot;DPanel&quot;, f ) -- Can be any panel, it will be stretched
local RightPanel = vgui.Create( &quot;DPanel&quot;, f )

local div = vgui.Create( &quot;DHorizontalDivider&quot;, f )
div:Dock( FILL ) -- Make the divider fill the space of the DFrame
div:SetLeft( LeftPanel ) -- Set what panel is in left side of the divider
div:SetRight( RightPanel )
div:SetDividerWidth( 4 ) -- Set the divider width. Default is 8
div:SetLeftMin( 20 ) -- Set the Minimum width of left side
div:SetRightMin( 20 )
div:SetLeftWidth( 50 ) -- Set the default left side width</code></pre>
