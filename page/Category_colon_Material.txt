<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a Material panel with TV static under a <a href="/page/Category_colon_DModelPanel" title="Category:DModelPanel">DModelPanel</a> with a spinning Garry's Mod logo model.
</p>
<pre><code data-language="lua">-- Background panel
BGPanel = vgui.Create(&quot;DPanel&quot;)
BGPanel:SetSize(400, 400)
BGPanel:Center()
BGPanel:SetBackgroundColor(Color(0, 0, 0, 255))

-- Material panel with TV static
local mat = vgui.Create(&quot;Material&quot;, BGPanel)
mat:SetPos(10, 10)
mat:SetSize(380, 380)
mat:SetMaterial(&quot;effects/tvscreen_noise002a&quot;)	-- Path to material VMT

-- Set this to false to enable material stretching
mat.AutoSize = false

-- Model panel for GMod Logo
local mdl = vgui.Create(&quot;DModelPanel&quot;, BGPanel)
mdl:SetPos(10, 10)
mdl:SetSize(380, 380)
mdl:SetModel(&quot;models/maxofs2d/logo_gmod_b.mdl&quot;)
mdl:SetCamPos(Vector(240, 0, 0))
mdl:SetLookAt(Vector(0, 0, 0))
mdl:SetFOV(40)

-- Spin faster
function mdl:LayoutEntity(ent)
	ent:SetAngles(Angle(0, RealTime()*100,  0))
end</code></pre>

<h1 class="first_example"> <span class="mw-headline" id="Examples_2">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example_2">Example</span></h2>
<p>Creates a custom material (using existing textures) of a Portal background with scrolling scan lines, and then displays that material in a Material panel in the center of the screen.
</p>
<pre><code data-language="lua">-- Background panel
BGPanel = vgui.Create(&quot;DPanel&quot;)
BGPanel:SetSize(720, 480)
BGPanel:Center()
BGPanel:SetBackgroundColor(Color(0, 0, 0, 255))

-- Material data for the scanline background
local matdata = {
	[&quot;$basetexture&quot;]=&quot;vgui/appchooser/background_portal_widescreen&quot;,
	[&quot;$texture2&quot;]=&quot;dev/dev_scanline&quot;,
	[&quot;Proxies&quot;]={
		[&quot;TextureScroll&quot;]={
			[&quot;texturescrollvar&quot;]=&quot;$texture2transform&quot;,
			[&quot;texturescrollrate&quot;]=.06,
			[&quot;texturescrollangle&quot;]=-90
		}
	}
}

-- Create material and recompute just in case
local portal_scanlines = CreateMaterial(&quot;PortalScanlines&quot;, &quot;UnlitTwoTexture&quot;, matdata)
portal_scanlines:Recompute()

-- Create material panel
local mat = vgui.Create(&quot;Material&quot;, BGPanel)
mat:SetPos(5, 5)
mat:SetSize(710, 470)

-- This has to be set manually since mat:SetMaterial only accepts string argument
mat.Material = portal_scanlines

-- Stretch to fit
mat.AutoSize = false</code></pre>
