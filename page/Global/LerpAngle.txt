<h1> <span class="mw-headline" id="Description">Description</span></h1>

<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Turns an entity 180 degrees uses lerp over ten seconds
</p>
<pre><code data-language="lua">local startAngle = Angle(0, 0, 0)
local endAngle = Angle(0, 180, 0)
local ratio = 0

timer.Create(&quot;Turn&quot;, 0.1, 10, function()
    ratio = ratio + 0.1
    entity:SetAngles(Lerp(ratio, startAngle, endAngle))
end)</code></pre>
