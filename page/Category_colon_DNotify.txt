<h1> <span class="mw-headline" id="DescriptionPanelPage">Description<span style="display: none;"><a href="/page/Form:PanelPage" title="Form:PanelPage">PanelPage</a></span></span></h1>
<p>A panel that fades its contents in and out once, like a notification.
</p>

<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a notification panel with a text label inside.
</p>
<pre><code data-language="lua">--Notification panel
local NotifyPanel = vgui.Create(&quot;DNotify&quot;)
NotifyPanel:SetPos(10, 5)
NotifyPanel:SetSize(200, 40)

-- Text label
local lbl = vgui.Create(&quot;DLabel&quot;, NotifyPanel)
lbl:Dock(FILL)
lbl:SetText(&quot;This is a notification.&quot;)
lbl:SetFont(&quot;GModNotify&quot;)
lbl:SetDark(true)

-- Add the label to the notification and begin fading
NotifyPanel:AddItem(lbl)</code></pre>
<p><br />
</p>
<h1 class="first_example"> <span class="mw-headline" id="Examples_2">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example_2">Example</span></h2>
<p>Creates a notification panel of Dr. Kleiner reminding the player to wear their HEV suit.
</p>
<pre><code data-language="lua">-- Notification panel
local NotifyPanel = vgui.Create(&quot;DNotify&quot;)
NotifyPanel:SetPos(15, 15)
NotifyPanel:SetSize(150, 210)

-- Gray background panel
local bg = vgui.Create(&quot;DPanel&quot;, NotifyPanel)
bg:Dock(FILL)
bg:SetBackgroundColor(Color(64, 64, 64))

-- Image of Dr. Kleiner (parented to background panel)
local img = vgui.Create(&quot;DImage&quot;, bg)
img:SetPos(11, 11)
img:SetSize(128, 128)
img:SetImage(&quot;entities/npc_kleiner.png&quot;)

-- A yellow label message (parented to background panel)
local lbl = vgui.Create(&quot;DLabel&quot;, bg)
lbl:SetPos(11, 136)
lbl:SetSize(128, 72)
lbl:SetText(&quot;Remember to wear your HEV suit!&quot;)
lbl:SetTextColor(Color(255, 200, 0))
lbl:SetFont(&quot;GModNotify&quot;)
lbl:SetWrap(true)

-- Add the background panel to the notification
NotifyPanel:AddItem(bg)</code></pre>
