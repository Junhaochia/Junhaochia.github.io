<h1> <span class="mw-headline" id="Description">Description</span></h1>
<p>Whenever the entity is removed, entityToRemove will be removed also.
</p>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_Entity" title="Category:Entity">Entity</a> entityToRemove</span>
</p>
<div style="margin-left: 32px;">
The entity to be removed</div>
</div>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a second chair in spawned jeeps that is removed when the jeep is removed
</p>
<pre><code data-language="lua">hook.Add(&quot;PlayerSpawnedVehicle&quot;, &quot;VehicleUpgrade&quot;, function(ply,vehicle)
	if vehicle:GetClass() == &quot;prop_vehicle_jeep&quot; then
		local seat = ents.Create( 'prop_vehicle_prisoner_pod' )
		seat:SetModel( &quot;models/nova/jeep_seat.mdl&quot; )
		seat:SetPos( vehicle:LocalToWorld(Vector(21,-32,18)) )
		seat:SetAngles( vehicle:LocalToWorldAngles(Angle(0,-3.5,0)) )
		seat:Spawn()
		seat:SetKeyValue( &quot;limitview&quot;, 0 )
		table.Merge( seat, { HandleAnimation = function(_,ply) return ply:SelectWeightedSequence( ACT_HL2MP_SIT ) end } )
		gamemode.Call( &quot;PlayerSpawnedVehicle&quot;, ply, seat )
		vehicle.PassengerSeat = seat
		vehicle:DeleteOnRemove(seat) //&lt;--
		constraint.Weld(seat, vehicle)
	end
end)</code></pre>
