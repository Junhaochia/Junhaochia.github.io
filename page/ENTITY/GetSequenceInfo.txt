<h1> <span class="mw-headline" id="Description">Description</span></h1>
<p>Returns a table of information about an entity's sequence.
</p>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_number" title="Category:number">number</a> sequenceId</span>
</p>
<div style="margin-left: 32px;">
The sequence id of the entity.</div>
</div>
<h1 class="first_ret"> <span class="mw-headline" id="Returns">Returns</span></h1>
<div class="return" style="counter-increment: returns;">
<p><span class="ret_number" style="counter-increment: returns -1;"></span> <span class="ret_chunk"><a href="/page/Category_colon_table" title="Category:table">table</a></span>
</p>
<div style="margin-left: 32px;">
Table of information about the entity's sequence, or nil is ID is out of range. See <a href="/page/Structures/SequenceInfo" title="Structures/SequenceInfo">SequenceInfo structure</a></div>
</div>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Draw each player's current sequence bounding box with sequence name, activity number, and activity name above their head.
</p>
<pre><code data-language="lua">function GM:PostPlayerDraw( ply )

	seqinfo = ply:GetSequenceInfo( ply:GetSequence() )
	seqinfo.player = ply
	
	render.DrawWireframeBox( ply:GetPos(), ply:GetAngles(), seqinfo.bbmin, seqinfo.bbmax, color_white, true )
	
end

function GM:HUDPaint()

	local seqinfo, textpos = nil, nil

	for p, ply in pairs(player.GetAll()) do
	
		seqinfo = ply:GetSequenceInfo(ply:GetSequence())
		textpos = ( ply:GetPos() + Vector( 0, 0, seqinfo.bbmax.z + 10 ) ):ToScreen()
	
		if ( textpos.visible ) then
			draw.SimpleText( seqinfo.label, &quot;GModNotify&quot;, textpos.x, textpos.y, color_white, TEXT_ALIGN_CENTER )
			draw.SimpleText( seqinfo.activity..&quot;: &quot;..seqinfo.activityname, &quot;GModNotify&quot;, textpos.x, textpos.y+20, color_white, TEXT_ALIGN_CENTER )
		end
	
	end

end</code></pre>
