<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Prints the MAT_ enum name for every prop on the map.
</p>
<pre><code data-language="lua">local function BackwardsEnums( enumname ) -- Helper function to build our table of values.
	local backenums = {}

	for k, v in pairs( _G ) do
		if type( k ) == &quot;string&quot; and string.find( k, &quot;^&quot;..enumname ) then
			backenums[ v ] = k
		end
	end

	return backenums
end

local MAT = BackwardsEnums( &quot;MAT_&quot; )

local validclasses = { prop_physics = true, prop_physics_multiplayer = true, prop_dynamic = true }
for k, v in pairs( ents.GetAll() ) do
	if validclasses[ v:GetClass() ] then
		print( v:GetModel(), MAT[ v:GetMaterialType() ] or &quot;UNKNOWN&quot; )
	end
end</code></pre>
