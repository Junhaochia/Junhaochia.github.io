<p>How this is supposed to work internally
</p>
<pre><code data-language="lua">ENT.Entities = {}

function ENT:IsTouchedBy( ent )
	return table.HasValue( self.Entities, ent )
end

function ENT:StartTouch( ent )
	if (&#160;!self:PassesTriggerFilters( ent ) ) then return end
	table.insert( self.Entities, ent )

	/* Code */
end

function ENT:Touch( ent )
	if (&#160;!self:PassesTriggerFilters( ent ) ) then return end
	if (&#160;!table.HasValue( ent ) ) then table.insert( self.Entities, ent ) end

	/* Code */
end

function ENT:EndTouch( ent )
	if (&#160;!self:IsTouchedBy( ent ) ) then return end
	table.RemoveByValue( self.Entities, ent )

	/* Code */
end</code></pre>
