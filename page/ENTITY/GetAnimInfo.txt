<h1> <span class="mw-headline" id="Description">Description</span></h1>
<p>Returns a table containing the number of frames, flags, name, and FPS of an entity's animation ID.
</p>
<table cellspacing="0" class="gmodwiki_note">
<tr>
<td>NOTE
</td>
<td>
<p>Animation ID is not the same as sequence ID.
</p>
</td></tr></table>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_number" title="Category:number">number</a> animIndex</span>
</p>
<div style="margin-left: 32px;">
The animation ID to look up</div>
</div>
<h1 class="first_ret"> <span class="mw-headline" id="Returns">Returns</span></h1>
<div class="return" style="counter-increment: returns;">
<p><span class="ret_number" style="counter-increment: returns -1;"></span> <span class="ret_chunk"><a href="/page/Category_colon_table" title="Category:table">table</a></span>
</p>
<div style="margin-left: 32px;">
Information about the animation, or nil if the index is out of bounds</div>
</div>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>A function that finds an entity sequence's corresponding animation and returns the animation info.
</p>
<pre><code data-language="lua">function GetAnimInfoSequence( ent, seq )

	if(&#160;!IsValid( ent ) ) then return nil end

	local seqname = ent:GetSequenceName( seq )
	
	if( seqname == &quot;Unknown&quot; ) then return nil end

	local info = nil
	local done = ent:GetAnimInfo(0).label	-- this is how we know when to stop
	local i = 1

	-- We don't want to increment too high or we will run into errors or possibly crashes
	while(i &lt; 1600) do    -- arbitrary failsafe
	
		info = ent:GetAnimInfo(i)
		
		if(string.find(info.label, &quot;@&quot;..seqname) or string.find(info.label, &quot;a_&quot;..seqname)) then
			return info
		end
		
		-- the first animation info is repeated when there are no animations left in the model
		if(info.label == done) then break end
		
		i = i + 1
		
	end

	return nil

end</code></pre>
