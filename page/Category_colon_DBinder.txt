<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a DBinder that prints to the player's chat when a new key is bound.
</p>
<pre><code data-language="lua">local frame = vgui.Create(&quot;DFrame&quot;)
frame:SetSize(250,100)
frame:Center()
frame:SetTitle(&quot;DBinder Example&quot;)
frame:MakePopup()

local binder = vgui.Create( &quot;DBinder&quot;, frame )
binder:SetSize( 200, 50 )
binder:SetPos( 25, 35 )

function binder:OnChange( num )
	LocalPlayer():ChatPrint(&quot;New bound key: &quot;..input.GetKeyName( num ))
end</code></pre>
