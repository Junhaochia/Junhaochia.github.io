<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a DHorizontalScroller with a bunch of <a href="/page/Category_colon_DImage" title="Category:DImage">DImages</a> attached to it.
</p>
<pre><code data-language="lua">local DFrame = vgui.Create( &quot;DFrame&quot; )
DFrame:SetTitle( &quot;DHorizontalScroller Example&quot; )
DFrame:SetSize( 500, 100 )
DFrame:Center()
DFrame:MakePopup()

local DHorizontalScroller = vgui.Create( &quot;DHorizontalScroller&quot;, DFrame )
DHorizontalScroller:Dock( FILL )
DHorizontalScroller:SetOverlap( -4 )

for i = 0, 16 do
	local DImage = vgui.Create( &quot;DImage&quot;, DHorizontalScroller )
	DImage:SetImage( &quot;scripted/breen_fakemonitor_1&quot; )
	DHorizontalScroller:AddPanel( DImage )
end</code></pre>

<h1 class="first_example"> <span class="mw-headline" id="Examples_2">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example_2">Example</span></h2>
<p>Creates a DHorizontalScroller with a bunch of <a href="/page/Category_colon_DImage" title="Category:DImage">DImages</a> attached to it and demonstrates how to color the left/right scroll buttons.
</p>
<pre><code data-language="lua">local DFrame = vgui.Create( &quot;DFrame&quot; )
DFrame:SetSize( 500, 100 )
DFrame:Center()
DFrame:MakePopup()
DFrame:SetTitle( &quot;DHorizontalScroller Example&quot; )
function DFrame:Paint( w, h )
	draw.RoundedBox( 0, 0, 0, w, h, Color( 0, 100, 100 ) )
end

local DHorizontalScroller = vgui.Create( &quot;DHorizontalScroller&quot;, DFrame )
DHorizontalScroller:Dock( FILL )
DHorizontalScroller:SetOverlap( -4 )

function DHorizontalScroller.btnLeft:Paint( w, h )
	draw.RoundedBox( 0, 0, 0, w, h, Color( 200, 100, 0 ) )
end
function DHorizontalScroller.btnRight:Paint( w, h )
	draw.RoundedBox( 0, 0, 0, w, h, Color( 0, 100, 200 ) )
end

for i = 0, 16 do
	local DImage = vgui.Create( &quot;DImage&quot;, DHorizontalScroller )
	DImage:SetImage( &quot;scripted/breen_fakemonitor_1&quot; )
	DHorizontalScroller:AddPanel( DImage )
end</code></pre>
