<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a menu with one option "Do you want to die?", which has two sub-options "Yes" and "No".
</p>
<pre><code data-language="lua">local parentMenu = DermaMenu()

local subMenu, parentMenuOption = parentMenu:AddSubMenu(&quot;Do you want to die?&quot;)
parentMenuOption:SetIcon(&quot;icon16/user_red.png&quot;)

local yesOption = subMenu:AddOption(&quot;Yes&quot;, function() LocalPlayer():ConCommand(&quot;kill&quot;) end)
yesOption:SetIcon(&quot;icon16/accept.png&quot;)

local noOption = subMenu:AddOption(&quot;No&quot;, function() print(&quot;You chose to live another day&quot;) end)
noOption:SetIcon(&quot;icon16/cross.png&quot;)

parentMenu:Open()</code></pre>
