<h1> <span class="mw-headline" id="DescriptionPanelPage">Description<span style="display: none;"><a href="/page/Form:PanelPage" title="Form:PanelPage">PanelPage</a></span></span></h1>

</p>

<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a DIconBrowser which, when an icon is clicked, prints said icon's directory.
</p>
<pre><code data-language="lua">local DermaPanel = vgui.Create( &quot;DFrame&quot; )
DermaPanel:SetPos( 100, 100 )
DermaPanel:SetSize( 300, 200 )
DermaPanel:SetTitle( &quot;DIconBrowserExample&quot; )
DermaPanel:SetVisible( true )
DermaPanel:SetDraggable( true )
DermaPanel:ShowCloseButton( true )
DermaPanel:MakePopup()

local IconBrowser = vgui.Create( &quot;DIconBrowser&quot;, DermaPanel )
IconBrowser:SetPos( 5, 30 )
IconBrowser:SetSize( 290, 265 )
IconBrowser.OnChange = function(self)
    chat.AddText( &quot;You selected: &quot; .. self:GetSelectedIcon())
end</code></pre>
