<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Example of styling a scrollbar
</p>
<pre><code data-language="lua">local DFrame = vgui.Create(&quot;DFrame&quot;)
DFrame:SetSize(500, 500)
DFrame:Center()
DFrame:MakePopup()
DFrame:SetTitle(&quot;Scrollbar Example&quot;)
function DFrame:Paint(w, h)
	draw.RoundedBox(0, 0, 0, w, h, Color(0, 100, 100))
end

local DScrollPanel = vgui.Create(&quot;DScrollPanel&quot;, DFrame)
DScrollPanel:SetSize(400, 250)
DScrollPanel:Center()

local sbar = DScrollPanel:GetVBar()
function sbar:Paint(w, h)
	draw.RoundedBox(0, 0, 0, w, h, Color(0, 0, 0, 100))
end
function sbar.btnUp:Paint(w, h)
	draw.RoundedBox(0, 0, 0, w, h, Color(200, 100, 0))
end
function sbar.btnDown:Paint(w, h)
	draw.RoundedBox(0, 0, 0, w, h, Color(200, 100, 0))
end
function sbar.btnGrip:Paint(w, h)
	draw.RoundedBox(0, 0, 0, w, h, Color(100, 200, 0))
end

local str = &quot;&quot;
for i = 1, 50 do str = str .. &quot;more space!\n&quot; end

local DLabel = vgui.Create(&quot;DLabel&quot;, DScrollPanel)
DLabel:SetText(str)
DLabel:Center()
DLabel:SizeToContents()</code></pre>
