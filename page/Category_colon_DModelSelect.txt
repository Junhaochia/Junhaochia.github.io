<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a DModelSelect within a <a href="/page/Category_colon_DFrame" title="Category:DFrame">DFrame</a> and adds four models.
</p>
<pre><code data-language="lua">-- Create a table of models
local models = {
	[&quot;models/props_c17/oildrum001_explosive.mdl&quot;] = {}, 
	[&quot;models/props_c17/oildrum001.mdl&quot;] = {}, 
	[&quot;models/props_junk/TrafficCone001a.mdl&quot;] = {},
	[&quot;models/props_c17/gravestone004a.mdl&quot;] = {}
}
	
local frame = vgui.Create(&quot;DFrame&quot;)
frame:SetSize(220, 220)
frame:SetTitle(&quot;DModelSelect Example&quot;)
frame:MakePopup()
frame:Center()

local mselect = vgui.Create( &quot;DModelSelect&quot;, frame ) 
mselect:SetModelList( models, &quot;&quot;, false, true )
mselect:SetSize(150, 150)
mselect:Center()</code></pre>
