<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Example showing the different types of format codes.
</p>
<pre><code data-language="lua">local s = &quot;Hello, world!&quot;
 
// string
print(string.format(&quot;here's a string:&#160;%s&quot;, s))
 
// string with quotes
print(string.format(&quot;here's a quoted string:&#160;%q&quot;, s))
 
// characters from numeric values
print(string.format(&quot;%c%c%c&quot;, 65, 66, 67))
 
// number with an exponent
print(string.format(&quot;%e,&#160;%E&quot;, math.pi, math.pi))
 
// float and compact float
print(string.format(&quot;%f,&#160;%G&quot;, math.pi, math.pi))
 
// signed, signed, and unsigned int
print(string.format(&quot;%d,&#160;%i,&#160;%u&quot;, -100, -100, -100))
 
// octal, hex, and uppercase hex
print(string.format(&quot;%o,&#160;%x,&#160;%X&quot;, -100, -100, -100))</code></pre>

</p>
