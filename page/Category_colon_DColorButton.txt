<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a DColorButton buton.
</p>
<pre><code data-language="lua">local frame = vgui.Create( &quot;DFrame&quot; )
frame:SetSize( 500, 500 )
frame:Center()
frame:MakePopup()

local DColorButton = vgui.Create( &quot;DColorButton&quot;, frame )
DColorButton:SetPos( 1, 28 )
DColorButton:SetSize( 100, 30 )
DColorButton:Paint( 100, 30 )
DColorButton:SetText( &quot;DColorButton&quot; )
DColorButton:SetColor( Color( 0, 110, 160 ) )
function DColorButton:DoClick() -- Callback inherited from DLabel, which is DColorButton's base
	print( &quot;I am clicked! My color is &quot;, self:GetColor() )
end</code></pre>
