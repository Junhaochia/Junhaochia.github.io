<h1> <span class="mw-headline" id="DescriptionPanelFunc">Description<span style="display: none;"><a href="/page/Form:PanelFunc" title="Form:PanelFunc">PanelFunc</a></span></span></h1>
<p>Sets the active tab of the <a href="/page/Category_colon_DPropertySheet" title="Category:DPropertySheet">DPropertySheet</a>.
</p>
<h1 class="first_arg"> <span class="mw-headline" id="Arguments">Arguments</span></h1>
<div class="argument">
<p><span class="arg_number"></span> <span class="arg_chunk"><a href="/page/Category_colon_Panel" title="Category:Panel">Panel</a> tab</span>
</p>
<div style="margin-left: 32px;">
<p>The <a href="/page/Category_colon_DTab" title="Category:DTab">DTab</a> to set active.
</p>
See <a href="/page/Category_colon_DPropertySheet" title="Category:DPropertySheet">DPropertySheet</a>:<a href="/page/DPropertySheet/GetItems" title="DPropertySheet/GetItems">GetItems</a></div>
</div>
<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Example of how you'd create and use this panel and set active tab.
</p>
<pre><code data-language="lua">local MainFrame = vgui.Create( &quot;DFrame&quot; )
MainFrame:SetSize( 500, 300 )
MainFrame:Center()
MainFrame:MakePopup()

local MainSheet = vgui.Create( &quot;DPropertySheet&quot;, MainFrame )
MainSheet:Dock( FILL )

local First_Panel = vgui.Create( &quot;DPanel&quot;, MainSheet )
First_Panel.Paint = function( self, w, h ) draw.RoundedBox( 4, 0, 0, w, h, Color( 0, 128, 255, self:GetAlpha() ) ) end
MainSheet:AddSheet( &quot;test&quot;, First_Panel, &quot;icon16/cross.png&quot; )

local Second_Panel = vgui.Create( &quot;DPanel&quot;, MainSheet )
Second_Panel.Paint = function( self, w, h ) draw.RoundedBox( 4, 0, 0, w, h, Color( 255, 128, 0, self:GetAlpha() ) ) end
MainSheet:AddSheet( &quot;test 2&quot;, Second_Panel, &quot;icon16/tick.png&quot; )

MainSheet:SetActiveTab( MainSheet:GetItems()[2].Tab ) --2 is a representation of the second sheet</code></pre>
