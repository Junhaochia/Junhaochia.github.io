<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Freezes an NPC for a period of time.
</p>
<pre><code data-language="lua">COND_NPC_UNFREEZE = 68	-- Delete this line when COND_ enums are added

function FreezeNPCTemporarily(npc, delay)
	
	if(!IsValid(npc)) then return end
	
	delay = delay or 1
	
	npc:SetSchedule(SCHED_NPC_FREEZE)

	timer.Simple(delay, function()
		if(IsValid(npc)) then npc:SetCondition(COND_NPC_UNFREEZE) end
	end)
	
end</code></pre>
