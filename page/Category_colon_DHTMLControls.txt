<h1 class="first_example"> <span class="mw-headline" id="Examples">Examples</span></h1>
<h2 class="examples_number"> <span class="mw-headline" id="Example">Example</span></h2>
<p>Creates a <a href="/page/Category_colon_DHTML" title="Category:DHTML">DHTML</a> window, controls for it, and links the two.
</p>
<pre><code data-language="lua">local frame = vgui.Create( &quot;DFrame&quot; ) -- Create a container for everything
frame:SetSize( 800, 600 )
frame:SetTitle( &quot;DHTMLControls Example&quot; )
frame:Center()
frame:MakePopup()

local window = vgui.Create( &quot;DHTML&quot;, frame ) -- Our DHTML window
window:SetSize( 750, 500 )
window:Center()

local ctrls = vgui.Create( &quot;DHTMLControls&quot;, frame) -- Navigation controls
ctrls:SetWide( 750 )
ctrls:SetPos( 25, 40 )
ctrls:SetHTML( window ) -- Links the controls to the DHTML window
ctrls.AddressBar:SetText( &quot;http://wiki.garrysmod.com&quot; ) -- Address bar isn't updated automatically

window:MoveBelow( ctrls ) -- Align the window to sit below the controls
window:OpenURL( &quot;http://wiki.garrysmod.com&quot; )</code></pre>
